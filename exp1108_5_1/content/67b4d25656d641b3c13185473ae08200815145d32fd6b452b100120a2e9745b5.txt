# signature: def postcondition(cost: List[List[int]], minimum_cost_symmetric_painting: int) -> bool
from typing import List

def postcondition(cost: List[List[int]], minimum_cost_symmetric_painting: int) -> bool:
    n = len(cost)
    
    # Check constraints
    if n < 2 or n % 2 != 0:
        raise ValueError("Invalid input")
    
    for i in range(n):
        if len(cost[i]) != 3:
            raise ValueError("Invalid input")
        for j in range(3):
            if not (0 <= cost[i][j] <= 10**5):
                raise ValueError("Invalid input")
    
    # Check if minimum_cost_symmetric_painting is non-negative
    if minimum_cost_symmetric_painting < 0:
        return False
    
    # Find all valid colorings and check if minimum_cost_symmetric_painting matches any valid cost
    # Since n can be up to 10^5, we need an efficient approach
    
    # For each pair of symmetric houses (i, n-1-i), they must have different colors
    # Also, adjacent houses must have different colors
    
    # We'll use dynamic programming to find the minimum cost
    # dp[i][c1][c2] = minimum cost for first i pairs where the i-th pair has colors c1 and c2
    
    # Initialize DP table
    INF = float('inf')
    
    # For the first pair (houses 0 and n-1)
    dp = [[INF for _ in range(3)] for _ in range(3)]
    for c1 in range(3):
        for c2 in range(3):
            if c1 != c2:  # Symmetric houses must have different colors
                dp[c1][c2] = cost[0][c1] + cost[n-1][c2]
    
    # Process remaining pairs
    for i in range(1, n//2):
        new_dp = [[INF for _ in range(3)] for _ in range(3)]
        
        for prev_c1 in range(3):
            for prev_c2 in range(3):
                if dp[prev_c1][prev_c2] == INF:
                    continue
                
                # Current pair: houses i and n-1-i
                for curr_c1 in range(3):
                    for curr_c2 in range(3):
                        # Check constraints:
                        # 1. Adjacent houses: house i-1 and house i must have different colors
                        if prev_c1 == curr_c1:
                            continue
                        # 2. Symmetric houses: house i and house n-1-i must have different colors
                        if curr_c1 == curr_c2:
                            continue
                        
                        total_cost = dp[prev_c1][prev_c2] + cost[i][curr_c1] + cost[n-1-i][curr_c2]
                        if total_cost < new_dp[curr_c1][curr_c2]:
                            new_dp[curr_c1][curr_c2] = total_cost
        
        dp = new_dp
    
    # Find the minimum cost from all valid configurations
    min_cost = INF
    for c1 in range(3):
        for c2 in range(3):
            min_cost = min(min_cost, dp[c1][c2])
    
    # Check if the given minimum_cost_symmetric_painting matches our calculated minimum
    if min_cost == INF:
        return False
    
    return abs(min_cost - minimum_cost_symmetric_painting) < 1e-9
