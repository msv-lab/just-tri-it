# signature: def minOperations_inv_k(min_operations_for_k_equal_subarrays: int, nums: List[int], x: int) -> int
from typing import List
import math

def minOperations_inv_k(min_operations_for_k_equal_subarrays: int, nums: List[int], x: int) -> int:
    if not nums or x < 2 or x > len(nums) or min_operations_for_k_equal_subarrays < 0:
        raise ValueError("Invalid input")
    
    n = len(nums)
    
    # Precompute prefix sums for efficient range sum calculation
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + nums[i]
    
    # Function to calculate operations needed for a subarray to become all equal to target
    def operations_for_subarray(start, target):
        # Operations = sum of absolute differences between elements and target
        # Since we can only increment/decrement by 1, cost = sum(|nums[i] - target|)
        # We can compute this efficiently using prefix sums and sorted values
        # But for simplicity, we'll compute directly for the subarray
        total = 0
        for i in range(start, start + x):
            total += abs(nums[i] - target)
        return total
    
    # Function to find min operations for exactly k subarrays
    def min_operations_for_k(k):
        # We need to select k non-overlapping subarrays of length x
        # This is a dynamic programming problem
        # dp[i][j] = min operations to select j subarrays from first i elements
        # But we need to optimize for large n
        
        # Alternative approach: sort all possible subarrays by their operation cost
        # and select the k cheapest non-overlapping ones
        
        # Generate all possible subarrays with their costs
        subarrays = []
        for i in range(n - x + 1):
            # For minimum operations, the optimal target is the median of the subarray
            sub = nums[i:i + x]
            sorted_sub = sorted(sub)
            median = sorted_sub[x // 2]
            cost = operations_for_subarray(i, median)
            subarrays.append((i, cost))
        
        # Sort subarrays by cost
        subarrays.sort(key=lambda x: x[1])
        
        # Greedily select k non-overlapping subarrays with minimum total cost
        selected = []
        total_cost = 0
        
        for idx, cost in subarrays:
            # Check if this subarray overlaps with any selected subarray
            overlap = False
            for sel_idx, _ in selected:
                if not (idx + x <= sel_idx or sel_idx + x <= idx):
                    overlap = True
                    break
            
            if not overlap:
                selected.append((idx, cost))
                total_cost += cost
                if len(selected) == k:
                    break
        
        return total_cost if len(selected) == k else float('inf')
    
    # Binary search for k
    left, right = 1, n // x
    answer = -1
    
    # Find the maximum k where min_operations <= min_operations_for_k_equal_subarrays
    while left <= right:
        mid = (left + right) // 2
        ops = min_operations_for_k(mid)
        
        if ops <= min_operations_for_k_equal_subarrays:
            answer = mid
            left = mid + 1
        else:
            right = mid - 1
    
    if answer == -1:
        raise ValueError("Invalid input")
    
    # Verify constraints
    if answer < 1 or answer > 15 or answer * x > n or answer * x < 2:
        raise ValueError("Invalid input")
    
    return answer
