# signature: def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool
from typing import List
from collections import defaultdict, deque

def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool:
    n = len(nums)
    
    # Validate input constraints
    if n < 2 or n > 50000:
        raise ValueError("Invalid input")
    if len(edges) != n - 1:
        raise ValueError("Invalid input")
    if len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    # Build adjacency list for the tree
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length < 1 or length > 1000:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Validate nums array
    if len(nums) != n:
        raise ValueError("Invalid input")
    for num in nums:
        if num < 0 or num > 50000:
            raise ValueError("Invalid input")
    
    # Build the tree structure with parent-child relationships
    parent = [-1] * n
    depth = [0] * n
    dist_from_root = [0] * n
    
    # BFS to assign parent-child relationships and calculate distances
    queue = deque([0])
    visited = [False] * n
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                dist_from_root[neighbor] = dist_from_root[node] + length
                queue.append(neighbor)
    
    # Find all special paths and track the longest ones
    max_path_length = 0
    min_nodes_for_max_length = float('inf')
    
    # For each node as the starting point, find special paths to its descendants
    for start in range(n):
        # DFS to find all special paths starting from 'start'
        stack = [(start, {nums[start]}, 0)]  # (current_node, values_seen, path_length)
        
        while stack:
            node, values, path_length = stack.pop()
            
            # Update maximum path length and minimum nodes
            if path_length > max_path_length:
                max_path_length = path_length
                min_nodes_for_max_length = depth[node] - depth[start] + 1
            elif path_length == max_path_length:
                nodes_count = depth[node] - depth[start] + 1
                if nodes_count < min_nodes_for_max_length:
                    min_nodes_for_max_length = nodes_count
            
            # Explore children
            for child, edge_length in graph[node]:
                if child == parent[node]:  # Skip parent
                    continue
                
                if nums[child] not in values:
                    new_values = values.copy()
                    new_values.add(nums[child])
                    new_path_length = path_length + edge_length
                    stack.append((child, new_values, new_path_length))
    
    # Handle case where no path was found (shouldn't happen as single node paths exist)
    if max_path_length == 0:
        max_path_length = 0
        min_nodes_for_max_length = 1
    
    # Check if the provided longest_special_path_info matches our calculation
    claimed_length, claimed_min_nodes = longest_special_path_info
    
    return claimed_length == max_path_length and claimed_min_nodes == min_nodes_for_max_length
