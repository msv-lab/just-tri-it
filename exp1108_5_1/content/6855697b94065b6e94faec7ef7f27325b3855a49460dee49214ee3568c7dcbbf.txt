# signature: def minOperations_sinv_k(min_operations_for_k_equal_subarrays: int, nums: List[int], x: int) -> list[int]
from typing import List
import math

def minOperations_sinv_k(min_operations_for_k_equal_subarrays: int, nums: List[int], x: int) -> List[int]:
    # Input validation
    if not nums or x < 2 or x > len(nums) or min_operations_for_k_equal_subarrays < 0:
        raise ValueError("Invalid input")
    
    n = len(nums)
    max_k = min(15, n // x)  # k can be at most 15 and k*x <= n
    
    # Precompute prefix sums for efficient range sum calculation
    prefix_sum = [0] * (n + 1)
    for i in range(n):
        prefix_sum[i + 1] = prefix_sum[i] + nums[i]
    
    # Precompute operations needed for each possible subarray
    # operations[i] = min operations to make subarray starting at i of length x equal
    operations = [0] * (n - x + 1)
    
    for i in range(n - x + 1):
        # Extract the subarray
        subarray = nums[i:i + x]
        # Sort to find median
        sorted_sub = sorted(subarray)
        median = sorted_sub[x // 2]
        
        # Calculate operations needed: sum of absolute differences from median
        total_ops = 0
        for num in subarray:
            total_ops += abs(num - median)
        operations[i] = total_ops
    
    # DP approach to find minimum operations for exactly k subarrays
    # dp[k][i] = min operations to choose k subarrays starting from position i
    max_possible_k = max_k
    
    # Initialize DP table
    dp = [[float('inf')] * (n + 1) for _ in range(max_possible_k + 1)]
    
    # Base case: 0 subarrays require 0 operations
    for i in range(n + 1):
        dp[0][i] = 0
    
    # Fill DP table
    for k_val in range(1, max_possible_k + 1):
        for i in range(n - k_val * x, -1, -1):
            # Option 1: Don't take current subarray starting at i
            if i + 1 <= n:
                dp[k_val][i] = min(dp[k_val][i], dp[k_val][i + 1])
            
            # Option 2: Take current subarray starting at i
            if i + x <= n and k_val - 1 >= 0:
                dp[k_val][i] = min(dp[k_val][i], 
                                 operations[i] + dp[k_val - 1][i + x])
    
    # Find all k where min operations equals target
    result = []
    for k_val in range(1, max_possible_k + 1):
        if k_val * x > n:
            continue
            
        min_ops = float('inf')
        for i in range(n - k_val * x + 1):
            min_ops = min(min_ops, dp[k_val][i])
        
        if min_ops == min_operations_for_k_equal_subarrays:
            result.append(k_val)
    
    return result
