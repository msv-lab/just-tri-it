# signature: def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool
from typing import List
from collections import defaultdict, deque

def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool:
    n = len(nums)
    
    # Validate input constraints
    if n < 2 or n > 50000:
        raise ValueError("Invalid input")
    if len(edges) != n - 1:
        raise ValueError("Invalid input")
    if len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    # Build adjacency list for the tree
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length < 1 or length > 1000:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Validate nums array
    if len(nums) != n:
        raise ValueError("Invalid input")
    if any(num < 0 or num > 50000 for num in nums):
        raise ValueError("Invalid input")
    
    # Build the tree structure (rooted at 0)
    parent = [-1] * n
    depth = [0] * n
    dist_from_root = [0] * n
    
    # BFS to assign parent and depth
    queue = deque([0])
    visited = [False] * n
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                dist_from_root[neighbor] = dist_from_root[node] + length
                queue.append(neighbor)
    
    # Find actual longest special path using DFS
    max_length = 0
    min_nodes_for_max_length = float('inf')
    
    def dfs(node, path_values, path_nodes, current_length):
        nonlocal max_length, min_nodes_for_max_length
        
        # Check if current path is special (all values unique)
        if len(path_values) == len(set(path_values)):
            # This is a valid special path
            if current_length > max_length:
                max_length = current_length
                min_nodes_for_max_length = len(path_nodes)
            elif current_length == max_length:
                min_nodes_for_max_length = min(min_nodes_for_max_length, len(path_nodes))
        
        # Explore children
        for child, edge_length in graph[node]:
            if child != parent[node]:  # Only go downward
                # Check if we can add this child to the path
                if nums[child] not in path_values:
                    # Add child to path and continue
                    path_values.append(nums[child])
                    path_nodes.append(child)
                    dfs(child, path_values, path_nodes, current_length + edge_length)
                    path_values.pop()
                    path_nodes.pop()
    
    # Start DFS from each node as potential starting point
    for start_node in range(n):
        dfs(start_node, [nums[start_node]], [start_node], 0)
    
    # Check if the provided longest_special_path_info matches our calculation
    claimed_length = longest_special_path_info[0]
    claimed_min_nodes = longest_special_path_info[1]
    
    return claimed_length == max_length and claimed_min_nodes == min_nodes_for_max_length
