# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict, deque

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    n = len(nums)
    
    # Validate inputs
    if n < 2 or len(edges) != n - 1:
        raise ValueError("Invalid input")
    
    for edge in edges:
        if len(edge) != 3:
            raise ValueError("Invalid input")
        u, v, length = edge
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
    
    if len(nums) != n:
        raise ValueError("Invalid input")
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Find the root and build parent-child relationships
    # We need to make it a rooted tree at node 0
    parent = [-1] * n
    children = defaultdict(list)
    edge_lengths = {}  # Store edge lengths from parent to child
    
    # BFS to build tree structure rooted at 0
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append(neighbor)
                edge_lengths[(node, neighbor)] = length
                queue.append(neighbor)
    
    # DFS to compute the longest special path
    # We'll track for each node: (max_length, min_nodes) for paths ending at that node
    # and also maintain a frequency map for values in the current path
    
    result = [0, 1]  # [max_length, min_nodes_for_max_length]
    
    def dfs(node, path_values, current_length):
        nonlocal result
        
        # Check if current node value is unique in the path
        if nums[node] in path_values:
            # Can't extend path further with this node
            # But the path up to parent is valid
            if current_length > result[0]:
                result[0] = current_length
                result[1] = len(path_values)
            elif current_length == result[0]:
                result[1] = min(result[1], len(path_values))
            return
        
        # Add current node to path
        path_values.add(nums[node])
        
        # Update result for current path
        if current_length > result[0]:
            result[0] = current_length
            result[1] = len(path_values)
        elif current_length == result[0]:
            result[1] = min(result[1], len(path_values))
        
        # Recursively process children
        for child in children[node]:
            edge_length = edge_lengths[(node, child)]
            dfs(child, path_values.copy(), current_length + edge_length)
    
    # Start DFS from root
    dfs(0, set(), 0)
    
    # Handle case where no path with positive length exists
    if result[0] == 0:
        # All single-node paths have length 0
        result[1] = 1
    
    return result
