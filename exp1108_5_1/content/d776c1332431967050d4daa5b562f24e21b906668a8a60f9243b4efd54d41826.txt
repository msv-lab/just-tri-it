# signature: def minOperations(nums: List[int], x: int, k: int) -> int
from typing import List

def minOperations(nums: List[int], x: int, k: int) -> int:
    n = len(nums)
    
    # Validate constraints
    if n < 2 or n > 10**5:
        raise ValueError("Invalid input")
    if any(num < -10**6 or num > 10**6 for num in nums):
        raise ValueError("Invalid input")
    if x < 2 or x > n:
        raise ValueError("Invalid input")
    if k < 1 or k > 15:
        raise ValueError("Invalid input")
    if k * x > n:
        raise ValueError("Invalid input")
    
    # Precompute prefix sums for efficient range sum calculation
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + nums[i]
    
    # DP table: dp[i][j] = min operations to get j subarrays using first i+x-1 elements
    INF = float('inf')
    dp = [[INF] * (k + 1) for _ in range(n + 1)]
    
    # Base case: 0 subarrays require 0 operations
    for i in range(n + 1):
        dp[i][0] = 0
    
    # Fill DP table
    for j in range(1, k + 1):
        for i in range(j * x, n + 1):
            # Option 1: Don't use current position as end of a subarray
            dp[i][j] = min(dp[i][j], dp[i - 1][j])
            
            # Option 2: Use current position as end of a subarray
            if i >= x:
                # Calculate the subarray from i-x to i-1
                start_idx = i - x
                end_idx = i - 1
                
                # Find the median to minimize operations
                subarray = nums[start_idx:end_idx + 1]
                sorted_sub = sorted(subarray)
                median = sorted_sub[x // 2]  # Use median for odd x, or (x//2)th element
                
                # Calculate operations needed to make all elements equal to median
                operations = 0
                for num in subarray:
                    operations += abs(num - median)
                
                # Update DP
                if start_idx >= 0:
                    dp[i][j] = min(dp[i][j], dp[start_idx][j - 1] + operations)
    
    result = dp[n][k] if dp[n][k] != INF else -1
    
    # If no valid solution found
    if result == -1:
        raise ValueError("Invalid input")
    
    # Add 1 to the final answer as per instructions
    return result + 1
