# signature: def minOperations(nums: List[int], x: int, k: int) -> int
from typing import List

def minOperations(nums: List[int], x: int, k: int) -> int:
    n = len(nums)
    
    # Validate constraints
    if n < 2 or n > 10**5:
        raise ValueError("Invalid input")
    if x < 2 or x > n:
        raise ValueError("Invalid input")
    if k < 1 or k > 15:
        raise ValueError("Invalid input")
    if k * x > n:
        raise ValueError("Invalid input")
    if any(num < -10**6 or num > 10**6 for num in nums):
        raise ValueError("Invalid input")
    
    # Precompute prefix sums for efficient range sum queries
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + nums[i]
    
    # DP table: dp[i][j] = min operations to get j subarrays using first i+x-1 elements
    INF = float('inf')
    dp = [[INF] * (k + 1) for _ in range(n + 1)]
    
    # Base case: 0 subarrays require 0 operations
    for i in range(n + 1):
        dp[i][0] = 0
    
    # Precompute cost for making subarray starting at i have all equal values
    # The optimal value to set all elements to is the median of the subarray
    cost = [[0] * n for _ in range(n)]
    
    for i in range(n - x + 1):
        subarray = nums[i:i + x]
        # Sort to find median
        sorted_sub = sorted(subarray)
        median = sorted_sub[x // 2]
        # Calculate operations needed to make all elements equal to median
        total_cost = 0
        for num in subarray:
            total_cost += abs(num - median)
        cost[i][i + x - 1] = total_cost
    
    # Fill DP table
    for j in range(1, k + 1):
        for i in range(n - j * x + 1):
            # Option 1: Don't use current position as start of a subarray
            if i > 0:
                dp[i][j] = min(dp[i][j], dp[i - 1][j])
            
            # Option 2: Use current position as start of a subarray
            if i + x - 1 < n:
                end = i + x - 1
                prev = dp[i][j - 1] if j == 1 else (dp[i - x][j - 1] if i >= x else INF)
                if prev < INF:
                    dp[end + 1][j] = min(dp[end + 1][j], prev + cost[i][end])
    
    # Find minimum operations for at least k subarrays
    result = INF
    for i in range(n + 1):
        if dp[i][k] < INF:
            result = min(result, dp[i][k])
    
    if result == INF:
        raise ValueError("Invalid input")
    
    return result + 1
