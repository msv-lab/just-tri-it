# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict, deque

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    n = len(nums)
    if n < 2:
        raise ValueError("Invalid input")
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Find root (node 0) and build parent-child relationships
    parent = [-1] * n
    children = defaultdict(list)
    edge_lengths = {}
    
    # BFS to build tree structure
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append(neighbor)
                edge_lengths[(node, neighbor)] = length
                queue.append(neighbor)
    
    # DFS to compute longest special paths
    max_length = 0
    min_nodes_for_max_length = float('inf')
    
    def dfs(node, current_path_values, current_path_length, current_node_count):
        nonlocal max_length, min_nodes_for_max_length
        
        # Check if current path is special (all values unique)
        if len(current_path_values) == len(set(current_path_values)):
            # Update global maximum
            if current_path_length > max_length:
                max_length = current_path_length
                min_nodes_for_max_length = current_node_count
            elif current_path_length == max_length:
                if current_node_count < min_nodes_for_max_length:
                    min_nodes_for_max_length = current_node_count
        
        # Explore children
        for child in children[node]:
            child_length = edge_lengths[(node, child)]
            
            # Check if adding child maintains uniqueness
            if nums[child] in current_path_values:
                # Can't extend path with this child, but the path ending at current node is valid
                # Check if current path itself is valid
                if len(current_path_values) == len(set(current_path_values)):
                    if current_path_length > max_length:
                        max_length = current_path_length
                        min_nodes_for_max_length = current_node_count
                    elif current_path_length == max_length:
                        if current_node_count < min_nodes_for_max_length:
                            min_nodes_for_max_length = current_node_count
            else:
                # Can extend path
                dfs(child, current_path_values + [nums[child]], 
                    current_path_length + child_length, current_node_count + 1)
    
    # Start DFS from each node as potential starting point
    for start_node in range(n):
        dfs(start_node, [nums[start_node]], 0, 1)
    
    # Handle case where no special path with positive length exists
    if max_length == 0:
        # Find all single-node paths (always special)
        min_nodes_for_max_length = 1
    
    return [max_length, min_nodes_for_max_length]
