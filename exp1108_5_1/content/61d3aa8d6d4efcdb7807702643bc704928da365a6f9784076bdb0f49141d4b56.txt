# signature: def postcondition(nums: List[int], x: int, k: int, min_operations_for_k_equal_subarrays: int) -> bool
from typing import List

def postcondition(nums: List[int], x: int, k: int, min_operations_for_k_equal_subarrays: int) -> bool:
    n = len(nums)
    
    # Validate constraints
    if not (2 <= n <= 10**5):
        raise ValueError("Invalid input")
    if not all(-10**6 <= num <= 10**6 for num in nums):
        raise ValueError("Invalid input")
    if not (2 <= x <= n):
        raise ValueError("Invalid input")
    if not (1 <= k <= 15):
        raise ValueError("Invalid input")
    if not (2 <= k * x <= n):
        raise ValueError("Invalid input")
    if min_operations_for_k_equal_subarrays < 0:
        raise ValueError("Invalid input")
    
    # Precompute prefix sums for efficient range sum calculation
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + nums[i]
    
    # DP table: dp[i][j] = min operations to get j valid subarrays using first i+x-1 elements
    # We use a large number as infinity
    INF = 10**18
    dp = [[INF] * (k + 1) for _ in range(n + 1)]
    
    # Base case: 0 subarrays require 0 operations
    for i in range(n + 1):
        dp[i][0] = 0
    
    # For each possible starting position of a subarray
    for i in range(n - x + 1):
        # Calculate cost to make this subarray have all equal elements
        # The optimal value to set all elements to is the median of the subarray
        # For cost minimization with absolute differences, median minimizes the sum
        subarray = nums[i:i + x]
        sorted_sub = sorted(subarray)
        median = sorted_sub[x // 2]
        
        cost = 0
        for num in subarray:
            cost += abs(num - median)
        
        # Update DP table
        for j in range(k, 0, -1):
            # If we don't use this subarray
            dp[i + 1][j] = min(dp[i + 1][j], dp[i][j])
            
            # If we use this subarray
            if dp[i][j - 1] != INF:
                dp[i + x][j] = min(dp[i + x][j], dp[i][j - 1] + cost)
    
    # Also consider the case where we don't use the last possible subarray
    for i in range(n - x + 1, n + 1):
        for j in range(k + 1):
            if i < n:
                dp[i + 1][j] = min(dp[i + 1][j], dp[i][j])
    
    # Find the minimum operations needed to get at least k subarrays
    min_ops_needed = INF
    for i in range(n + 1):
        min_ops_needed = min(min_ops_needed, dp[i][k])
    
    # Return True if the given value is >= the minimum operations needed
    return min_operations_for_k_equal_subarrays >= min_ops_needed
