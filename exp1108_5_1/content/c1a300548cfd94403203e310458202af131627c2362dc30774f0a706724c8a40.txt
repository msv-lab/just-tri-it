from typing import List
from collections import defaultdict, deque

def longestSpecialPath_inv_nums(longest_special_path_info: List[int], edges: List[List[int]]) -> List[int]:
    if not longest_special_path_info or len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    target_length, target_min_nodes = longest_special_path_info
    
    if target_length < 0 or target_min_nodes < 1:
        raise ValueError("Invalid input")
    
    n = len(edges) + 1
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Find parent-child relationships by BFS from root 0
    parent = [-1] * n
    children = defaultdict(list)
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, _ in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append(neighbor)
                queue.append(neighbor)
    
    # Handle the case where target_length is 0 (path of length 0)
    if target_length == 0:
        # All nodes can have the same value, creating no special path of length > 0
        return [0] * n
    
    # We need to create a path of exactly target_length nodes with unique values
    # The minimum number of nodes in the longest special path is target_min_nodes
    
    # Strategy: Create a chain of target_min_nodes nodes with unique values
    # and ensure no other path has more unique values
    
    nums = [0] * n
    
    # Find a path from root to leaf with at least target_min_nodes nodes
    path_nodes = []
    
    def dfs_find_path(node, path):
        nonlocal path_nodes
        if path_nodes:
            return
        path.append(node)
        if len(path) >= target_min_nodes:
            path_nodes = path[:]
            return
        for child in children[node]:
            dfs_find_path(child, path)
            if path_nodes:
                return
        path.pop()
    
    dfs_find_path(0, [])
    
    if not path_nodes:
        # If we can't find a path long enough, it's impossible
        raise ValueError("Invalid input")
    
    # Assign unique values to the path nodes
    used_values = set()
    for i, node in enumerate(path_nodes):
        nums[node] = i
        used_values.add(i)
    
    # Fill remaining nodes with values that will break uniqueness
    # Use a value that already appears in the path to ensure no longer paths
    next_val = target_min_nodes
    for i in range(n):
        if nums[i] == 0 and i not in path_nodes:
            # Use a value that repeats to break potential longer paths
            nums[i] = 0  # Reuse value 0 to break uniqueness
    
    return nums
