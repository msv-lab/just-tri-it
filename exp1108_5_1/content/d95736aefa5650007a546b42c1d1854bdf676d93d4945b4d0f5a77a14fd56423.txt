# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    n = len(nums)
    if n < 2 or len(edges) != n - 1:
        raise ValueError("Invalid input")
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # DFS to find parent-child relationships and build tree structure
    parent = [-1] * n
    depth = [0] * n
    dist_from_root = [0] * n
    
    stack = [0]
    parent[0] = -1
    depth[0] = 0
    dist_from_root[0] = 0
    
    # Build the tree with DFS
    children = defaultdict(list)
    while stack:
        node = stack.pop()
        for neighbor, length in graph[node]:
            if neighbor != parent[node]:
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                dist_from_root[neighbor] = dist_from_root[node] + length
                children[node].append((neighbor, length))
                stack.append(neighbor)
    
    # Result variables: [max_length, min_nodes_for_max_length]
    result = [0, 1]  # At minimum, a single node path has length 0 and 1 node
    
    # We'll use DFS to traverse and track the path
    def dfs(node, path_values, path_nodes, current_length):
        nonlocal result
        
        # Check if current path is valid (all unique values)
        if len(path_values) == len(set(path_values)):
            # Update result if we found a longer path
            if current_length > result[0]:
                result = [current_length, len(path_nodes)]
            elif current_length == result[0]:
                # Same length, update min nodes
                result[1] = min(result[1], len(path_nodes))
        
        # Explore children
        for child, length in children[node]:
            # Add child to path
            path_values.append(nums[child])
            path_nodes.append(child)
            
            # Recursively explore
            dfs(child, path_values, path_nodes, current_length + length)
            
            # Backtrack
            path_values.pop()
            path_nodes.pop()
    
    # Start DFS from each node as potential starting point
    for start_node in range(n):
        dfs(start_node, [nums[start_node]], [start_node], 0)
    
    return result
