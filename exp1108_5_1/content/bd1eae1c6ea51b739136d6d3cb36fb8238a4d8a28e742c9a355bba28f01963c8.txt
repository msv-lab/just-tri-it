# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict, deque

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    n = len(nums)
    
    # Validate inputs
    if n < 2:
        raise ValueError("Invalid input")
    if len(edges) != n - 1:
        raise ValueError("Invalid input")
    for edge in edges:
        if len(edge) != 3:
            raise ValueError("Invalid input")
        if edge[0] < 0 or edge[0] >= n or edge[1] < 0 or edge[1] >= n:
            raise ValueError("Invalid input")
        if edge[2] <= 0:
            raise ValueError("Invalid input")
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Find root (node 0) and build parent-child relationships
    parent = [-1] * n
    depth = [0] * n
    children = defaultdict(list)
    edge_length = defaultdict(int)
    
    # BFS to build tree structure
    queue = deque([0])
    visited = [False] * n
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                children[node].append(neighbor)
                edge_length[(node, neighbor)] = length
                edge_length[(neighbor, node)] = length
                queue.append(neighbor)
    
    # DFS to compute longest special paths
    max_length = 0
    min_nodes_for_max_length = float('inf')
    
    def dfs(node, current_path_values, current_path_length, current_nodes_count):
        nonlocal max_length, min_nodes_for_max_length
        
        # Check if current path is valid (all values are unique)
        if len(current_path_values) == len(set(current_path_values)):
            # Update global maximum
            if current_path_length > max_length:
                max_length = current_path_length
                min_nodes_for_max_length = current_nodes_count
            elif current_path_length == max_length:
                if current_nodes_count < min_nodes_for_max_length:
                    min_nodes_for_max_length = current_nodes_count
        
        # Explore children
        for child in children[node]:
            child_value = nums[child]
            edge_len = edge_length[(node, child)]
            
            # Continue path to child
            new_path_values = current_path_values + [child_value]
            dfs(child, new_path_values, current_path_length + edge_len, current_nodes_count + 1)
            
            # Start new path from child
            dfs(child, [child_value], 0, 1)
    
    # Start DFS from root with all possible starting points
    for i in range(n):
        dfs(i, [nums[i]], 0, 1)
    
    return [max_length, min_nodes_for_max_length]
