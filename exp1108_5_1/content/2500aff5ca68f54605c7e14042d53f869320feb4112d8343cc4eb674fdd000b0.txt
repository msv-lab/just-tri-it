# signature: def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool
from typing import List
from collections import defaultdict, deque

def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool:
    n = len(nums)
    
    # Validate input constraints
    if n < 2 or n > 50000:
        raise ValueError("Invalid input")
    if len(edges) != n - 1:
        raise ValueError("Invalid input")
    for edge in edges:
        if len(edge) != 3:
            raise ValueError("Invalid input")
        u, v, length = edge
        if u < 0 or u >= n or v < 0 or v >= n or length < 1 or length > 1000:
            raise ValueError("Invalid input")
    if len(nums) != n:
        raise ValueError("Invalid input")
    for num in nums:
        if num < 0 or num > 50000:
            raise ValueError("Invalid input")
    if len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    claimed_length, claimed_min_nodes = longest_special_path_info
    if claimed_length < 0 or claimed_min_nodes < 1:
        raise ValueError("Invalid input")
    
    # Build the tree as an adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Build parent-child relationships (root at 0)
    parent = [-1] * n
    children = defaultdict(list)
    edge_lengths = {}
    
    # BFS to establish parent-child relationships
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append(neighbor)
                edge_lengths[(node, neighbor)] = length
                edge_lengths[(neighbor, node)] = length
                queue.append(neighbor)
    
    # DFS to find all special paths
    max_length = 0
    min_nodes_for_max = float('inf')
    
    def dfs(node, current_path_values, current_path_nodes, current_length):
        nonlocal max_length, min_nodes_for_max
        
        # Check if current path is special (all unique values)
        if len(current_path_values) == len(set(current_path_values)):
            path_length = current_length
            num_nodes = len(current_path_nodes)
            
            if path_length > max_length:
                max_length = path_length
                min_nodes_for_max = num_nodes
            elif path_length == max_length:
                if num_nodes < min_nodes_for_max:
                    min_nodes_for_max = num_nodes
        
        # Continue DFS to children
        for child in children[node]:
            child_value = nums[child]
            edge_len = edge_lengths[(node, child)]
            
            # Create new sets for the path to child
            new_path_values = current_path_values + [child_value]
            new_path_nodes = current_path_nodes + [child]
            new_length = current_length + edge_len
            
            dfs(child, new_path_values, new_path_nodes, new_length)
    
    # Start DFS from each node as potential starting point
    for start_node in range(n):
        dfs(start_node, [nums[start_node]], [start_node], 0)
    
    # Check if the claimed values match the computed values
    return claimed_length == max_length and claimed_min_nodes == min_nodes_for_max
