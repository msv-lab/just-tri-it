# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    n = len(nums)
    if n < 2 or len(edges) != n - 1:
        raise ValueError("Invalid input")
    
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    parent = [-1] * n
    depth = [0] * n
    dist_from_root = [0] * n
    
    stack = [0]
    parent[0] = 0
    while stack:
        node = stack.pop()
        for neighbor, length in graph[node]:
            if parent[neighbor] == -1:
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                dist_from_root[neighbor] = dist_from_root[node] + length
                stack.append(neighbor)
    
    max_length = 0
    min_nodes = float('inf')
    
    for i in range(n):
        for j in range(i, n):
            if i == j:
                path_length = 0
                num_nodes = 1
                if path_length > max_length:
                    max_length = path_length
                    min_nodes = num_nodes
                elif path_length == max_length:
                    min_nodes = min(min_nodes, num_nodes)
                continue
            
            path_set = set()
            unique = True
            u, v = i, j
            
            temp_u, temp_v = u, v
            while depth[temp_u] > depth[temp_v]:
                if nums[temp_u] in path_set:
                    unique = False
                    break
                path_set.add(nums[temp_u])
                temp_u = parent[temp_u]
            
            if not unique:
                continue
                
            while depth[temp_v] > depth[temp_u]:
                if nums[temp_v] in path_set:
                    unique = False
                    break
                path_set.add(nums[temp_v])
                temp_v = parent[temp_v]
            
            if not unique:
                continue
                
            while temp_u != temp_v:
                if nums[temp_u] in path_set:
                    unique = False
                    break
                path_set.add(nums[temp_u])
                if nums[temp_v] in path_set:
                    unique = False
                    break
                path_set.add(nums[temp_v])
                temp_u = parent[temp_u]
                temp_v = parent[temp_v]
            
            if not unique:
                continue
                
            if nums[temp_u] in path_set:
                unique = False
            
            if unique:
                path_length = dist_from_root[i] + dist_from_root[j] - 2 * dist_from_root[temp_u]
                num_nodes = depth[i] + depth[j] - 2 * depth[temp_u] + 1
                
                if path_length > max_length:
                    max_length = path_length
                    min_nodes = num_nodes
                elif path_length == max_length:
                    min_nodes = min(min_nodes, num_nodes)
    
    if max_length == 0:
        min_nodes = 1
    
    result = [max_length, min_nodes]
    result[0] += 1
    result[1] += 1
    return result
