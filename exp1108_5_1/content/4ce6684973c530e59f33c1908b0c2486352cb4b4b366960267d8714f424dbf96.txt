# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict, deque

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    n = len(nums)
    
    # 验证输入
    if n < 2 or len(edges) != n - 1:
        raise ValueError("Invalid input")
    
    for edge in edges:
        if len(edge) != 3:
            raise ValueError("Invalid input")
        u, v, length = edge
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
    
    if len(nums) != n:
        raise ValueError("Invalid input")
    
    # 构建邻接表
    graph = defaultdict(list)
    for u, v, length in edges:
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # 找到根节点（0）并构建树结构
    parent = [-1] * n
    depth = [0] * n
    dist_from_root = [0] * n
    children = defaultdict(list)
    
    # BFS 构建树结构
    queue = deque([0])
    visited = [False] * n
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                dist_from_root[neighbor] = dist_from_root[node] + length
                children[node].append((neighbor, length))
                queue.append(neighbor)
    
    # 全局变量存储结果
    max_length = 0
    min_nodes_for_max_length = float('inf')
    
    def dfs(node):
        nonlocal max_length, min_nodes_for_max_length
        
        # 存储从当前节点开始的路径信息
        # 每个元素是 (路径长度, 节点数, 当前路径上的值集合)
        paths = []
        
        for child, edge_length in children[node]:
            child_paths = dfs(child)
            for path_length, node_count, value_set in child_paths:
                # 检查当前节点的值是否与路径中的值冲突
                if nums[node] in value_set:
                    continue
                # 添加当前节点到路径
                new_length = path_length + edge_length
                new_node_count = node_count + 1
                new_value_set = value_set | {nums[node]}
                
                paths.append((new_length, new_node_count, new_value_set))
                
                # 更新全局结果
                if new_length > max_length:
                    max_length = new_length
                    min_nodes_for_max_length = new_node_count
                elif new_length == max_length:
                    if new_node_count < min_nodes_for_max_length:
                        min_nodes_for_max_length = new_node_count
        
        # 添加单节点路径（当前节点自身）
        single_path = (0, 1, {nums[node]})
        paths.append(single_path)
        
        # 更新全局结果（单节点路径）
        if 0 > max_length:
            max_length = 0
            min_nodes_for_max_length = 1
        elif 0 == max_length:
            if 1 < min_nodes_for_max_length:
                min_nodes_for_max_length = 1
        
        # 合并来自不同子树的路径
        for i in range(len(paths)):
            for j in range(i + 1, len(paths)):
                len1, cnt1, set1 = paths[i]
                len2, cnt2, set2 = paths[j]
                
                # 检查值是否唯一
                if set1 & set2:
                    continue
                
                # 合并路径（通过当前节点连接）
                total_length = len1 + len2
                total_nodes = cnt1 + cnt2 - 1  # 减去重复的当前节点
                
                if total_length > max_length:
                    max_length = total_length
                    min_nodes_for_max_length = total_nodes
                elif total_length == max_length:
                    if total_nodes < min_nodes_for_max_length:
                        min_nodes_for_max_length = total_nodes
        
        return paths
    
    # 从根节点开始DFS
    dfs(0)
    
    # 处理没有找到任何路径的情况（理论上不会发生，因为至少单节点路径存在）
    if max_length == 0 and min_nodes_for_max_length == float('inf'):
        min_nodes_for_max_length = 1
    
    return [max_length, min_nodes_for_max_length]
