# signature: def minOperations(nums: List[int], x: int, k: int) -> int
from typing import List

def minOperations(nums: List[int], x: int, k: int) -> int:
    n = len(nums)
    
    # Input validation
    if not (2 <= n <= 10**5):
        raise ValueError("Invalid input")
    if not all(-10**6 <= num <= 10**6 for num in nums):
        raise ValueError("Invalid input")
    if not (2 <= x <= n):
        raise ValueError("Invalid input")
    if not (1 <= k <= 15):
        raise ValueError("Invalid input")
    if not (2 <= k * x <= n):
        raise ValueError("Invalid input")
    
    # Precompute prefix sums for efficient range sum queries
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + nums[i]
    
    # DP table: dp[i][j] = min operations to get j subarrays using first i+x-1 elements
    INF = float('inf')
    dp = [[INF] * (k + 1) for _ in range(n + 1)]
    dp[0][0] = 0
    
    for i in range(n - x + 1):
        for j in range(k + 1):
            if dp[i][j] == INF:
                continue
            
            # Option 1: Skip current position
            dp[i + 1][j] = min(dp[i + 1][j], dp[i][j])
            
            # Option 2: Start a subarray at position i (if we have enough remaining elements)
            if j < k and i + x <= n:
                # Calculate the target value that minimizes operations for this subarray
                subarray_sum = prefix[i + x] - prefix[i]
                # The optimal target is the median (for odd x) or either of the two middle values (for even x)
                # For simplicity, we'll use the median approach
                sub = nums[i:i + x]
                sub_sorted = sorted(sub)
                
                # For odd length, median is the middle element
                # For even length, either of the two middle elements works
                median = sub_sorted[x // 2]
                
                # Calculate operations needed to make all elements equal to median
                operations = 0
                for num in sub:
                    operations += abs(num - median)
                
                dp[i + x][j + 1] = min(dp[i + x][j + 1], dp[i][j] + operations)
    
    result = INF
    for i in range(n + 1):
        result = min(result, dp[i][k])
    
    if result == INF:
        raise ValueError("Invalid input")
    
    return result + 1
