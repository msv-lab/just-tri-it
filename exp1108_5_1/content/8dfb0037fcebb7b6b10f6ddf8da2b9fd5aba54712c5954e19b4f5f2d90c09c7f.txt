# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    n = len(nums)
    if n < 2:
        raise ValueError("Invalid input")
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Find root (node 0) and build parent-child relationships
    parent = [-1] * n
    children = defaultdict(list)
    edge_lengths = {}
    
    stack = [0]
    visited = [False] * n
    visited[0] = True
    
    while stack:
        node = stack.pop()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append(neighbor)
                edge_lengths[(node, neighbor)] = length
                stack.append(neighbor)
    
    # Result storage: [max_length, min_nodes_for_max_length]
    result = [0, 0]
    
    def dfs(node):
        nonlocal result
        
        # For each child, we'll get the best paths ending at that child
        child_paths = []
        
        for child in children[node]:
            child_res = dfs(child)
            child_paths.append(child_res)
        
        # Paths starting and ending at current node
        current_paths = [([nums[node]], 0, 1)]  # (values, total_length, node_count)
        
        # Combine with child paths
        for child_idx, child in enumerate(children[node]):
            edge_len = edge_lengths[(node, child)]
            
            for values, length, count in child_paths[child_idx]:
                # Check if we can extend this path with current node
                if nums[node] not in values:
                    new_values = values + [nums[node]]
                    new_length = length + edge_len
                    new_count = count + 1
                    current_paths.append((new_values, new_length, new_count))
        
        # Update global result
        for values, length, count in current_paths:
            if length > result[0]:
                result[0] = length
                result[1] = count
            elif length == result[0] and count < result[1]:
                result[1] = count
        
        # Return the best paths ending at current node (for parent to use)
        # We need to return paths that can be extended upward
        return_paths = []
        
        for values, length, count in current_paths:
            # We can return any valid path ending at this node
            return_paths.append((values, length, count))
        
        # Also include single-node path
        return_paths.append(([nums[node]], 0, 1))
        
        return return_paths
    
    # Start DFS from root
    dfs(0)
    
    # Handle case where no path with positive length was found
    if result[0] == 0:
        # Find the path with minimum nodes (should be 1 for single node)
        result[1] = 1
    
    return result
