# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict, deque

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    n = len(nums)
    
    # Validate inputs
    if n < 2 or len(edges) != n - 1:
        raise ValueError("Invalid input")
    
    for edge in edges:
        if len(edge) != 3:
            raise ValueError("Invalid input")
        u, v, length = edge
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
    
    if len(nums) != n:
        raise ValueError("Invalid input")
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Build tree structure with parent-child relationships (root at 0)
    parent = [-1] * n
    children = defaultdict(list)
    edge_lengths = {}  # Store edge lengths between parent and child
    
    # BFS to establish parent-child relationships
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append(neighbor)
                edge_lengths[(node, neighbor)] = length
                queue.append(neighbor)
    
    # DFS to compute longest special paths
    max_length = 0
    min_nodes_for_max_length = float('inf')
    
    def dfs(node, path_values, path_length, node_count):
        nonlocal max_length, min_nodes_for_max_length
        
        current_value = nums[node]
        
        # Check if current value already exists in path
        if current_value in path_values:
            # Can't extend path further with this value
            if path_length > max_length:
                max_length = path_length
                min_nodes_for_max_length = node_count
            elif path_length == max_length:
                min_nodes_for_max_length = min(min_nodes_for_max_length, node_count)
            return
        
        # Add current value to path
        path_values.add(current_value)
        
        # Update max for current path (path ending at this node)
        if path_length > max_length:
            max_length = path_length
            min_nodes_for_max_length = node_count
        elif path_length == max_length:
            min_nodes_for_max_length = min(min_nodes_for_max_length, node_count)
        
        # Continue DFS to children
        for child in children[node]:
            edge_len = edge_lengths[(node, child)]
            dfs(child, path_values.copy(), path_length + edge_len, node_count + 1)
    
    # Start DFS from all nodes as potential starting points
    for start_node in range(n):
        dfs(start_node, set(), 0, 1)
    
    # Handle case where no path was found (shouldn't happen per constraints)
    if max_length == 0:
        min_nodes_for_max_length = 1
    
    return [max_length, min_nodes_for_max_length]
