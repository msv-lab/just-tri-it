from typing import List
from collections import defaultdict, deque

def longestSpecialPath_inv_nums(longest_special_path_info: List[int], edges: List[List[int]]) -> List[int]:
    if not longest_special_path_info or len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    target_length, target_min_nodes = longest_special_path_info
    
    if target_length < 0 or target_min_nodes < 1:
        raise ValueError("Invalid input")
    
    n = len(edges) + 1
    
    # Build the tree
    graph = defaultdict(list)
    for u, v, length in edges:
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Find parent-child relationships using BFS from root 0
    parent = [-1] * n
    children = defaultdict(list)
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, _ in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append(neighbor)
                queue.append(neighbor)
    
    # Handle the case where we need a path of length 0 (single node)
    if target_length == 0:
        nums = [0] * n
        return nums
    
    # For paths with length >= 1, we need to construct a path with target_length + 1 nodes
    # (since path length is measured in edges, but we need unique values for nodes)
    path_node_count = target_length + 1
    
    if path_node_count > n:
        raise ValueError("Invalid input")
    
    # Initialize nums with distinct values
    nums = [0] * n
    
    # If we need exactly 1 node in the path (length 0)
    if path_node_count == 1:
        return nums
    
    # Construct a path with path_node_count nodes having unique values
    # We'll create a simple increasing sequence along a path from root
    
    # Find a path from root with enough nodes
    def find_path(node, depth, max_depth, current_path):
        if depth == max_depth:
            return current_path
        for child in children[node]:
            path = find_path(child, depth + 1, max_depth, current_path + [child])
            if path:
                return path
        return None
    
    # Try to find a path starting from root with path_node_count nodes
    special_path = find_path(0, 1, path_node_count, [0])
    
    if not special_path:
        # If we can't find a path of required length, try to build one with available nodes
        # For simplicity, we'll assign unique values to first path_node_count nodes in BFS order
        bfs_order = []
        queue = deque([0])
        visited = [False] * n
        visited[0] = True
        
        while queue:
            node = queue.popleft()
            bfs_order.append(node)
            for child in children[node]:
                if not visited[child]:
                    visited[child] = True
                    queue.append(child)
        
        special_path = bfs_order[:path_node_count]
    
    # Assign unique values to nodes in the special path
    for i, node in enumerate(special_path):
        nums[node] = i
    
    # Fill remaining nodes with values that won't create longer special paths
    # Use values that repeat to ensure no longer unique paths exist
    next_val = path_node_count
    for i in range(n):
        if nums[i] == 0 and i not in special_path:
            nums[i] = next_val % (path_node_count + 1)
            next_val += 1
    
    return nums
