# signature: def minOperations(nums: List[int], x: int, k: int) -> int
from typing import List

def minOperations(nums: List[int], x: int, k: int) -> int:
    n = len(nums)
    
    # Validate constraints
    if n < 2 or n > 10**5:
        raise ValueError("Invalid input")
    if x < 2 or x > n:
        raise ValueError("Invalid input")
    if k < 1 or k > 15:
        raise ValueError("Invalid input")
    if k * x > n:
        raise ValueError("Invalid input")
    if any(num < -10**6 or num > 10**6 for num in nums):
        raise ValueError("Invalid input")
    
    # Precompute prefix sums for efficient range sum calculation
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + nums[i]
    
    # Precompute cost for making all elements in a subarray equal to a target value
    # For subarray starting at index i, the optimal target is the median
    # But since we need k non-overlapping subarrays, we'll use dynamic programming
    
    # Calculate cost for each possible subarray
    # subarray_cost[i] = minimum operations to make subarray starting at i of length x equal
    subarray_cost = [0] * (n - x + 1)
    
    for i in range(n - x + 1):
        # Extract the subarray
        sub = nums[i:i + x]
        # Sort to find median
        sorted_sub = sorted(sub)
        # For odd length, median is middle element; for even, either middle works
        median = sorted_sub[x // 2]
        # Calculate cost to make all elements equal to median
        cost = 0
        for num in sub:
            cost += abs(num - median)
        subarray_cost[i] = cost
    
    # Now we need to select k non-overlapping subarrays with minimum total cost
    # Use dynamic programming: dp[i][j] = min cost to select j subarrays starting from position i
    # But k is small (<=15), so we can use a different approach
    
    # We'll use memoization for DP
    memo = {}
    
    def dfs(pos, remaining):
        if remaining == 0:
            return 0
        if pos > n - x:
            return float('inf')
        
        if (pos, remaining) in memo:
            return memo[(pos, remaining)]
        
        # Option 1: Skip current position
        skip = dfs(pos + 1, remaining)
        
        # Option 2: Take subarray starting at pos
        take = subarray_cost[pos] + dfs(pos + x, remaining - 1)
        
        memo[(pos, remaining)] = min(skip, take)
        return memo[(pos, remaining)]
    
    result = dfs(0, k)
    
    # According to the problem, add 1 to the final answer
    return result + 1
