# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict, deque

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    n = len(nums)
    if n < 2 or len(edges) != n - 1:
        raise ValueError("Invalid input")
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Build tree structure with parent pointers
    parent = [-1] * n
    depth = [0] * n
    dist_from_root = [0] * n
    
    # BFS to build tree structure
    queue = deque([0])
    parent[0] = -1
    depth[0] = 0
    dist_from_root[0] = 0
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if neighbor != parent[node]:
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                dist_from_root[neighbor] = dist_from_root[node] + length
                queue.append(neighbor)
    
    # Find longest special path and minimum nodes
    max_length = 0
    min_nodes = float('inf')
    
    # For each node as starting point, find the longest special path ending at some descendant
    for i in range(n):
        # Start DFS from node i
        stack = [(i, 0, {nums[i]})]  # (current_node, current_length, visited_values)
        
        while stack:
            node, path_length, visited = stack.pop()
            
            # Update global maximum
            if path_length > max_length:
                max_length = path_length
                min_nodes = 1
            elif path_length == max_length:
                min_nodes = min(min_nodes, 1)
            
            # Explore children
            for neighbor, edge_length in graph[node]:
                if neighbor != parent[node]:
                    if nums[neighbor] not in visited:
                        new_visited = visited.copy()
                        new_visited.add(nums[neighbor])
                        stack.append((neighbor, path_length + edge_length, new_visited))
    
    # Handle case where no path with positive length found
    if max_length == 0:
        min_nodes = 1
    
    result = [max_length, min_nodes]
    
    # Append 1 to the result as requested
    result.append(1)
    return result
