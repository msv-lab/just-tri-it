# signature: def minOperations(nums: List[int], x: int, k: int) -> int
from typing import List

def minOperations(nums: List[int], x: int, k: int) -> int:
    n = len(nums)
    
    # Input validation
    if n < 2 or n > 10**5:
        raise ValueError("Invalid input")
    if not all(-10**6 <= num <= 10**6 for num in nums):
        raise ValueError("Invalid input")
    if not (2 <= x <= n):
        raise ValueError("Invalid input")
    if not (1 <= k <= 15):
        raise ValueError("Invalid input")
    if k * x > n:
        raise ValueError("Invalid input")
    
    # Precompute prefix sums for efficient range sum calculation
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + nums[i]
    
    # DP state: dp[i][j] = min operations to have j valid subarrays considering first i positions
    INF = float('inf')
    dp = [[INF] * (k + 1) for _ in range(n + 1)]
    dp[0][0] = 0
    
    for i in range(n + 1):
        for j in range(k + 1):
            if dp[i][j] == INF:
                continue
            
            # Option 1: Skip current position
            if i + 1 <= n:
                dp[i + 1][j] = min(dp[i + 1][j], dp[i][j])
            
            # Option 2: Start a new subarray of length x at position i
            if j + 1 <= k and i + x <= n:
                # For subarray [i, i+x-1], we want all elements equal to minimize operations
                # The optimal value is the median (for odd x) or average of two medians (for even x)
                # But since we can only change by 1, we need to find the value that minimizes sum of absolute differences
                
                # Extract the subarray
                subarray = nums[i:i + x]
                sorted_sub = sorted(subarray)
                
                # For minimizing sum of absolute differences, the median is optimal
                median = sorted_sub[x // 2]
                
                # Calculate operations needed
                operations = 0
                for num in subarray:
                    operations += abs(num - median)
                
                dp[i + x][j + 1] = min(dp[i + x][j + 1], dp[i][j] + operations)
    
    result = dp[n][k]
    if result == INF:
        raise ValueError("Invalid input")
    
    return result + 1
