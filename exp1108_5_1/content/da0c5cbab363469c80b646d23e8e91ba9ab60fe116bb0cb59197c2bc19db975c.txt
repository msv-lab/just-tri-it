# signature: def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool
from typing import List
from collections import defaultdict, deque

def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool:
    n = len(nums)
    
    # Validate input constraints
    if n < 2 or n > 50000:
        raise ValueError("Invalid input")
    if len(edges) != n - 1:
        raise ValueError("Invalid input")
    for edge in edges:
        if len(edge) != 3:
            raise ValueError("Invalid input")
        u, v, length = edge
        if u < 0 or u >= n or v < 0 or v >= n or length < 1 or length > 1000:
            raise ValueError("Invalid input")
    if len(nums) != n:
        raise ValueError("Invalid input")
    for num in nums:
        if num < 0 or num > 50000:
            raise ValueError("Invalid input")
    if len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    # Build the tree as an adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Build the tree structure with parent-child relationships (rooted at 0)
    parent = [-1] * n
    depth = [0] * n
    dist_from_root = [0] * n
    children = defaultdict(list)
    
    # BFS to establish parent-child relationships and distances
    queue = deque([0])
    visited = [False] * n
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                dist_from_root[neighbor] = dist_from_root[node] + length
                children[node].append((neighbor, length))
                queue.append(neighbor)
    
    # Function to check if a path is special (all node values are unique)
    def is_special_path(path_nodes):
        values = set()
        for node in path_nodes:
            if nums[node] in values:
                return False
            values.add(nums[node])
        return True
    
    # Find all special paths and their lengths
    special_paths = []
    
    # We'll use DFS to explore all paths from each node
    def dfs(node, current_path, current_length):
        # Check if current path is special
        if is_special_path(current_path):
            path_length = current_length
            if current_path:
                # Calculate the actual path length (sum of edge lengths)
                if len(current_path) > 1:
                    path_length = dist_from_root[current_path[-1]] - dist_from_root[current_path[0]]
                else:
                    path_length = 0
                
                special_paths.append((path_length, len(current_path)))
            
            # Extend the path
            for child, edge_length in children[node]:
                dfs(child, current_path + [child], current_length + edge_length)
    
    # Start DFS from each node
    for i in range(n):
        dfs(i, [i], 0)
    
    if not special_paths:
        # If no special paths found, check if the claimed info is for empty case
        claimed_length, claimed_min_nodes = longest_special_path_info
        if claimed_length == 0 and claimed_min_nodes == 1:
            return True
        return False
    
    # Find the actual longest special path length
    max_length = max(length for length, _ in special_paths)
    
    # Find all paths with maximum length
    max_length_paths = [(length, node_count) for length, node_count in special_paths if length == max_length]
    
    if not max_length_paths:
        return False
    
    # Find the minimum number of nodes among all paths with maximum length
    min_nodes = min(node_count for _, node_count in max_length_paths)
    
    # Check if the claimed info matches
    claimed_length, claimed_min_nodes = longest_special_path_info
    
    return claimed_length == max_length and claimed_min_nodes == min_nodes
