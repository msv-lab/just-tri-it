# signature: def postcondition(nums: List[int], x: int, k: int, min_operations_for_k_equal_subarrays: int) -> bool
from typing import List

def postcondition(nums: List[int], x: int, k: int, min_operations_for_k_equal_subarrays: int) -> bool:
    n = len(nums)
    
    # Validate constraints
    if n < 2 or n > 10**5:
        raise ValueError("Invalid input")
    if any(num < -10**6 or num > 10**6 for num in nums):
        raise ValueError("Invalid input")
    if x < 2 or x > n:
        raise ValueError("Invalid input")
    if k < 1 or k > 15:
        raise ValueError("Invalid input")
    if k * x < 2 or k * x > n:
        raise ValueError("Invalid input")
    if min_operations_for_k_equal_subarrays < 0:
        raise ValueError("Invalid input")
    
    # Precompute prefix sums for efficient range sum calculation
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + nums[i]
    
    # DP state: dp[i][j] = minimum operations to get j valid subarrays using first i elements
    # We use a large number as infinity
    INF = 10**18
    dp = [[INF] * (k + 1) for _ in range(n + 1)]
    dp[0][0] = 0
    
    for i in range(n + 1):
        for j in range(k + 1):
            if dp[i][j] == INF:
                continue
                
            # Option 1: Skip current element (don't start a subarray here)
            if i + 1 <= n:
                dp[i + 1][j] = min(dp[i + 1][j], dp[i][j])
            
            # Option 2: Start a new subarray of size x at position i
            if j + 1 <= k and i + x <= n:
                # Calculate cost to make all elements in subarray nums[i:i+x] equal
                # The optimal target value is the median for odd x, or any value between
                # the two middle values for even x. For simplicity, we'll use the mean
                # which gives the minimum sum of absolute deviations
                subarray = nums[i:i + x]
                subarray_sorted = sorted(subarray)
                
                # For sum of absolute deviations, median minimizes the cost
                median = subarray_sorted[x // 2]
                
                # Calculate the actual cost
                cost = 0
                for num in subarray:
                    cost += abs(num - median)
                
                dp[i + x][j + 1] = min(dp[i + x][j + 1], dp[i][j] + cost)
    
    # Find the minimum operations needed to get at least k subarrays
    min_required = INF
    for i in range(n + 1):
        min_required = min(min_required, dp[i][k])
    
    # Return True if the given value is at least the minimum required
    return min_operations_for_k_equal_subarrays >= min_required
