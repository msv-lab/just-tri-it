# signature: def minOperations(nums: List[int], x: int, k: int) -> int
from typing import List

def minOperations(nums: List[int], x: int, k: int) -> int:
    if not nums or x <= 0 or k <= 0 or k * x > len(nums):
        raise ValueError("Invalid input")
    
    n = len(nums)
    
    # Precompute prefix sums for efficient range sum calculation
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + nums[i]
    
    # DP table: dp[i][j] = min operations to get j subarrays using first i+x-1 elements
    # Initialize with infinity
    INF = float('inf')
    dp = [[INF] * (k + 1) for _ in range(n + 1)]
    
    # Base case: 0 subarrays require 0 operations
    for i in range(n + 1):
        dp[i][0] = 0
    
    # Fill DP table
    for j in range(1, k + 1):
        for i in range(j * x, n + 1):
            # Option 1: Don't use current position as end of a subarray
            dp[i][j] = min(dp[i][j], dp[i - 1][j])
            
            # Option 2: Use positions [i-x, i-1] as a subarray
            if i >= x:
                # Calculate cost to make all elements in this subarray equal
                # The optimal value is the median (for odd x) or either of the two middle values (for even x)
                start, end = i - x, i - 1
                subarray = nums[start:end + 1]
                sorted_sub = sorted(subarray)
                
                # For odd x, median is at position x//2
                # For even x, both middle values give same cost
                median = sorted_sub[x // 2]
                
                # Calculate operations needed
                cost = 0
                for num in subarray:
                    cost += abs(num - median)
                
                # Update DP
                dp[i][j] = min(dp[i][j], dp[i - x][j - 1] + cost)
    
    result = dp[n][k]
    return int(result) if result != INF else 0
