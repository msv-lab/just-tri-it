from typing import List
from collections import defaultdict, deque

def longestSpecialPath_inv_nums(longest_special_path_info: List[int], edges: List[List[int]]) -> List[int]:
    if len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    target_length, target_min_nodes = longest_special_path_info
    
    if target_length < 1 or target_min_nodes < 1 or target_min_nodes > target_length:
        raise ValueError("Invalid input")
    
    n = len(edges) + 1
    
    # Build the tree
    graph = defaultdict(list)
    for u, v, length in edges:
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Find parent-child relationships by doing BFS from root 0
    parent = [-1] * n
    children = defaultdict(list)
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append(neighbor)
                queue.append(neighbor)
    
    # Strategy: Create a path of length target_length with target_min_nodes unique values
    # We'll create a path from root downwards with the required properties
    
    # Find a chain of nodes from root with length >= target_length
    # We can use DFS to find the longest chain from root
    def dfs_longest_chain(node):
        if not children[node]:
            return [node]
        longest = []
        for child in children[node]:
            chain = dfs_longest_chain(child)
            if len(chain) > len(longest):
                longest = chain
        return [node] + longest
    
    longest_chain = dfs_longest_chain(0)
    
    if len(longest_chain) < target_length:
        raise ValueError("Invalid input")
    
    # Take the first target_length nodes from the longest chain
    path_nodes = longest_chain[:target_length]
    
    # Assign values to create the special path
    nums = [0] * n
    
    # For nodes in our target path, assign unique values
    for i, node in enumerate(path_nodes):
        nums[node] = i
    
    # For nodes not in the path, assign values that won't create longer paths
    # We can assign values that are already used in the path to ensure they don't extend it
    for i in range(n):
        if nums[i] == 0 and i != 0:  # Skip root (already assigned 0) and assigned nodes
            nums[i] = 0  # Use a value that's already in the path
    
    # Verify our construction meets the requirements
    # The path from path_nodes[0] to path_nodes[target_length-1] should have target_length unique values
    # and target_min_nodes nodes
    
    # Since we're using consecutive integers 0,1,...,target_length-1 for the path,
    # all values are unique, so the path has exactly target_length unique values
    # and target_min_nodes nodes (which equals target_length in this case)
    
    # If target_min_nodes < target_length, we need to repeat some values
    if target_min_nodes < target_length:
        # We need to repeat some values in the path to reduce the number of unique nodes
        # while maintaining the path length
        repeats_needed = target_length - target_min_nodes
        
        # Replace some values in the path with already used values
        for i in range(1, repeats_needed + 1):
            if target_length - i >= 0:
                nums[path_nodes[target_length - i]] = nums[path_nodes[target_min_nodes - 1]]
    
    return nums
