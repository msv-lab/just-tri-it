# signature: def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool
from typing import List
from collections import defaultdict, deque

def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool:
    n = len(nums)
    
    # Validate input constraints
    if n < 2 or n > 50000:
        raise ValueError("Invalid input")
    if len(edges) != n - 1:
        raise ValueError("Invalid input")
    if len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    # Build adjacency list for the tree
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length < 1 or length > 1000:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Validate nums array
    if len(nums) != n:
        raise ValueError("Invalid input")
    for num in nums:
        if num < 0 or num > 50000:
            raise ValueError("Invalid input")
    
    # BFS to build parent-child relationships and distances from root
    parent = [-1] * n
    dist_from_root = [0] * n
    children = defaultdict(list)
    
    queue = deque([0])
    visited = [False] * n
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                dist_from_root[neighbor] = dist_from_root[node] + length
                children[node].append(neighbor)
                queue.append(neighbor)
    
    # DFS to find actual longest special path
    def dfs(node, current_path_values, current_path_nodes):
        nonlocal max_length, min_nodes_count
        
        # Add current node to path
        current_path_values.add(nums[node])
        current_path_nodes.append(node)
        
        # Check if current path is special (all unique values)
        if len(current_path_values) == len(current_path_nodes):
            # Calculate path length
            path_length = dist_from_root[current_path_nodes[-1]] - dist_from_root[current_path_nodes[0]]
            
            # Update max_length and min_nodes_count
            if path_length > max_length:
                max_length = path_length
                min_nodes_count = len(current_path_nodes)
            elif path_length == max_length:
                if len(current_path_nodes) < min_nodes_count:
                    min_nodes_count = len(current_path_nodes)
        
        # Continue DFS to children
        for child in children[node]:
            dfs(child, current_path_values.copy(), current_path_nodes.copy())
    
    # Initialize tracking variables
    max_length = 0
    min_nodes_count = float('inf')
    
    # Start DFS from each node as potential starting point
    for start_node in range(n):
        dfs(start_node, set(), [])
    
    # Handle case where no special path was found (should at least have single node paths)
    if max_length == 0:
        # Find all single node paths (length 0)
        single_node_paths = 0
        for i in range(n):
            single_node_paths += 1
        
        if single_node_paths > 0:
            max_length = 0
            min_nodes_count = 1
    
    # Check if the provided longest_special_path_info matches our calculation
    provided_length = longest_special_path_info[0]
    provided_min_nodes = longest_special_path_info[1]
    
    return provided_length == max_length and provided_min_nodes == min_nodes_count
