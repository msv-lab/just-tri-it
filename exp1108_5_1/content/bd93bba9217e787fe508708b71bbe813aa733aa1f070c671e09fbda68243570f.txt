from typing import List
from collections import defaultdict, deque

def longestSpecialPath_inv_nums(longest_special_path_info: List[int], edges: List[List[int]]) -> List[int]:
    if not longest_special_path_info or len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    target_length, target_min_nodes = longest_special_path_info
    
    if target_length < 0 or target_min_nodes < 1:
        raise ValueError("Invalid input")
    
    n = len(edges) + 1
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Build tree structure with parent-child relationships (root at 0)
    children = defaultdict(list)
    parent = [-1] * n
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append((neighbor, length))
                queue.append(neighbor)
    
    # Case 1: If target_length is 0, we need a path with 1 node (minimum)
    if target_length == 0:
        if target_min_nodes != 1:
            raise ValueError("Invalid input")
        # All nodes can have the same value (e.g., 0)
        return [0] * n
    
    # Case 2: If we need a special path of length L with minimum M nodes
    # We'll construct a path with exactly target_min_nodes nodes that has unique values
    # and has total length >= target_length
    
    # Find a suitable path in the tree
    path_nodes = []
    path_length = 0
    
    # Try to find a path starting from root with exactly target_min_nodes nodes
    def dfs_find_path(node, current_path, current_length):
        nonlocal path_nodes, path_length
        
        if len(path_nodes) == target_min_nodes:
            return
        
        current_path.append(node)
        
        # If we have exactly target_min_nodes nodes, check if this path works
        if len(current_path) == target_min_nodes:
            # Calculate total length of this path
            total_len = 0
            for i in range(len(current_path) - 1):
                # Find the edge length between current_path[i] and current_path[i+1]
                for child, length in children[current_path[i]]:
                    if child == current_path[i + 1]:
                        total_len += length
                        break
            
            if total_len >= target_length:
                path_nodes = current_path.copy()
                path_length = total_len
            current_path.pop()
            return
        
        # Continue DFS
        for child, length in children[node]:
            dfs_find_path(child, current_path, current_length + length)
            if path_nodes:
                break
        
        current_path.pop()
    
    # Try to find a suitable path
    dfs_find_path(0, [], 0)
    
    if not path_nodes:
        # If we couldn't find a path with exactly target_min_nodes nodes that meets length requirement,
        # try to find any path that meets the requirements
        # For simplicity, we'll create a chain from root with increasing values
        path_nodes = list(range(min(target_min_nodes, n)))
        path_length = 0
        for i in range(len(path_nodes) - 1):
            for child, length in children[path_nodes[i]]:
                if child == path_nodes[i + 1]:
                    path_length += length
                    break
        
        # If the natural path doesn't meet length requirement, we need to adjust
        if path_length < target_length:
            # This is a simplified approach - in practice, you'd need more sophisticated tree traversal
            raise ValueError("Cannot construct valid nums for given constraints")
    
    # Assign values to nodes
    nums = [0] * n
    
    # Assign unique values to path nodes
    for i, node in enumerate(path_nodes):
        nums[node] = i  # This ensures all values in the path are unique
    
    # Fill remaining nodes with values that won't create longer special paths
    # Use values that repeat to prevent creating longer unique-value paths
    next_val = len(path_nodes)
    for i in range(n):
        if nums[i] == 0 and i != path_nodes[0]:  # Node not in our special path
            nums[i] = next_val % (len(path_nodes) + 1)
            next_val += 1
    
    return nums
