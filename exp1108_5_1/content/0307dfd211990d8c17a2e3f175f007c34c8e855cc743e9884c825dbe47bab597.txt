# signature: def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool
from typing import List
from collections import defaultdict, deque

def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool:
    n = len(nums)
    
    # Validate input constraints
    if n < 2 or n > 50000:
        raise ValueError("Invalid input")
    
    if len(edges) != n - 1:
        raise ValueError("Invalid input")
    
    for edge in edges:
        if len(edge) != 3:
            raise ValueError("Invalid input")
        u, v, length = edge
        if u < 0 or u >= n or v < 0 or v >= n or length < 1 or length > 1000:
            raise ValueError("Invalid input")
    
    if len(nums) != n:
        raise ValueError("Invalid input")
    
    for num in nums:
        if num < 0 or num > 50000:
            raise ValueError("Invalid input")
    
    if len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    claimed_length = longest_special_path_info[0]
    claimed_min_nodes = longest_special_path_info[1]
    
    if claimed_length < 0 or claimed_min_nodes < 1:
        raise ValueError("Invalid input")
    
    # Build the tree
    graph = defaultdict(list)
    for u, v, length in edges:
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Build parent-child relationships for the tree rooted at 0
    parent = [-1] * n
    depth = [0] * n
    dist_from_root = [0] * n
    
    # BFS to build tree structure
    queue = deque([0])
    parent[0] = -1
    depth[0] = 0
    dist_from_root[0] = 0
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if neighbor != parent[node]:
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                dist_from_root[neighbor] = dist_from_root[node] + length
                queue.append(neighbor)
    
    # Find all special paths and track the longest ones
    max_length = 0
    min_nodes_for_max_length = float('inf')
    
    # For each node as the starting point, find special paths
    for start in range(n):
        # DFS to find special paths starting from 'start'
        stack = [(start, 0, {nums[start]})]  # (current_node, current_length, value_set)
        
        while stack:
            node, path_length, value_set = stack.pop()
            
            # Update max_length and min_nodes_for_max_length
            if path_length > max_length:
                max_length = path_length
                min_nodes_for_max_length = len(value_set)
            elif path_length == max_length:
                min_nodes_for_max_length = min(min_nodes_for_max_length, len(value_set))
            
            # Explore children
            for child, edge_length in graph[node]:
                if child != parent[node] and nums[child] not in value_set:
                    new_value_set = value_set.copy()
                    new_value_set.add(nums[child])
                    stack.append((child, path_length + edge_length, new_value_set))
    
    # Check if the claimed values match the actual values
    return claimed_length == max_length and claimed_min_nodes == min_nodes_for_max_length
