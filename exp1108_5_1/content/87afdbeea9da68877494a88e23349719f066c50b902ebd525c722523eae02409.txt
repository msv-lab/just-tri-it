# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
# test name: test_large_edge_weights
def test_large_edge_weights():
    """Test with larger edge weights to ensure proper accumulation"""
    edges = [[0,1,1000],[1,2,500],[1,3,750],[2,4,300]]
    nums = [10,20,30,40,50]  # All unique values
    result = longestSpecialPath(edges, nums)
    # Path 0->1->3 has length 1750, path 0->1->2->4 has length 1800
    assert result == [1800, 4]
