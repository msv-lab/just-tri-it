# signature: def minOperations(nums: List[int], x: int, k: int) -> int
# test name: test_floating_point_median_requires_rounding
def test_floating_point_median_requires_rounding():
    nums = [1, 2, 4, 5]
    x = 2
    k = 2
    result = minOperations(nums, x, k)
    # Pairs: (1,2) -> both 1 or both 2 (cost=1)
    # (4,5) -> both 4 or both 5 (cost=1)
    # Total cost = 2
    assert result == 2
