# signature: def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool
from typing import List
from collections import defaultdict, deque

def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool:
    n = len(nums)
    
    # Validate input constraints
    if n < 2 or n > 50000:
        raise ValueError("Invalid input")
    if len(edges) != n - 1:
        raise ValueError("Invalid input")
    if len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    # Build the tree as an adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length < 1 or length > 1000:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Validate nums values
    for num in nums:
        if num < 0 or num > 50000:
            raise ValueError("Invalid input")
    
    # Build parent-child relationships (tree rooted at 0)
    parent = [-1] * n
    depth = [0] * n
    dist_from_root = [0] * n
    
    # BFS to establish parent-child relationships and distances
    queue = deque([0])
    visited = [False] * n
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                dist_from_root[neighbor] = dist_from_root[node] + length
                queue.append(neighbor)
    
    # Find all special paths and track the longest ones
    max_length = 0
    min_nodes_for_max_length = float('inf')
    
    # For each node as the starting point, find special paths to descendants
    for start in range(n):
        # DFS to find all special paths starting from 'start'
        stack = [(start, 0, {nums[start]})]  # (current_node, path_length, unique_values)
        
        while stack:
            node, path_length, unique_values = stack.pop()
            
            # Update max_length and min_nodes_for_max_length
            if len(unique_values) == path_length + 1:  # All values are unique
                if path_length > max_length:
                    max_length = path_length
                    min_nodes_for_max_length = path_length + 1
                elif path_length == max_length:
                    min_nodes_for_max_length = min(min_nodes_for_max_length, path_length + 1)
            
            # Explore children
            for neighbor, edge_length in graph[node]:
                if neighbor != parent[node]:  # Only go downward in the tree
                    if nums[neighbor] not in unique_values:
                        new_unique_values = unique_values | {nums[neighbor]}
                        new_path_length = path_length + edge_length
                        stack.append((neighbor, new_path_length, new_unique_values))
    
    # Check if the provided longest_special_path_info matches our calculation
    claimed_length, claimed_min_nodes = longest_special_path_info
    
    # Handle edge case where no special path exists (shouldn't happen as single node paths are valid)
    if max_length == 0:
        return claimed_length == 0 and claimed_min_nodes == 1
    
    return claimed_length == max_length and claimed_min_nodes == min_nodes_for_max_length
