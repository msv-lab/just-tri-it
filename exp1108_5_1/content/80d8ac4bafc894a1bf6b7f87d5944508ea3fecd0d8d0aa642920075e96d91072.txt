# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    n = len(nums)
    if n < 2 or len(edges) != n - 1:
        raise ValueError("Invalid input")
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Find root (0) and build parent-child relationships
    parent = [-1] * n
    depth = [0] * n
    dist_from_root = [0] * n
    
    stack = [0]
    parent[0] = -1
    depth[0] = 0
    dist_from_root[0] = 0
    
    while stack:
        node = stack.pop()
        for neighbor, length in graph[node]:
            if neighbor != parent[node]:
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                dist_from_root[neighbor] = dist_from_root[node] + length
                stack.append(neighbor)
    
    # Precompute LCA data structures
    LOG = (n).bit_length()
    up = [[-1] * n for _ in range(LOG)]
    
    # Initialize first parent
    for i in range(n):
        up[0][i] = parent[i]
    
    # Binary lifting
    for i in range(1, LOG):
        for j in range(n):
            if up[i-1][j] != -1:
                up[i][j] = up[i-1][up[i-1][j]]
    
    def get_lca(a, b):
        if depth[a] < depth[b]:
            a, b = b, a
        
        # Lift a to same depth as b
        diff = depth[a] - depth[b]
        for i in range(LOG):
            if diff & (1 << i):
                a = up[i][a]
        
        if a == b:
            return a
        
        # Lift both until their parents are the same
        for i in range(LOG-1, -1, -1):
            if up[i][a] != up[i][b]:
                a = up[i][a]
                b = up[i][b]
        
        return up[0][a]
    
    def get_distance(u, v):
        lca = get_lca(u, v)
        return dist_from_root[u] + dist_from_root[v] - 2 * dist_from_root[lca]
    
    # Find all valid paths with unique values
    max_length = 0
    min_nodes_for_max_length = float('inf')
    
    # For each node as start, DFS to find paths with unique values
    visited = [False] * n
    
    def dfs(node, current_set, current_length, node_count):
        nonlocal max_length, min_nodes_for_max_length
        
        # Check if current path is valid (all values unique)
        if len(current_set) == node_count:  # All values are unique
            if current_length > max_length:
                max_length = current_length
                min_nodes_for_max_length = node_count
            elif current_length == max_length:
                min_nodes_for_max_length = min(min_nodes_for_max_length, node_count)
        
        for neighbor, edge_length in graph[node]:
            if not visited[neighbor] and depth[neighbor] > depth[node]:  # Go downward only
                if nums[neighbor] in current_set:
                    # Can't extend this path
                    continue
                
                visited[neighbor] = True
                current_set.add(nums[neighbor])
                dfs(neighbor, current_set, current_length + edge_length, node_count + 1)
                current_set.remove(nums[neighbor])
                visited[neighbor] = False
    
    # Try each node as starting point
    for start in range(n):
        visited = [False] * n
        visited[start] = True
        current_set = {nums[start]}
        dfs(start, current_set, 0, 1)
    
    # Handle case where no path with more than 1 node is found
    if max_length == 0:
        # Single node paths have length 0
        # Find the minimum node count among all single node paths
        min_nodes = 1
        return [0, min_nodes]
    
    return [max_length, min_nodes_for_max_length]
