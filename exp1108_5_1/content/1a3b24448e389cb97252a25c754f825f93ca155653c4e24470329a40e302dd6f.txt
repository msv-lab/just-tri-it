from typing import List
from collections import defaultdict, deque

def longestSpecialPath_inv_nums(longest_special_path_info: List[int], edges: List[List[int]]) -> List[int]:
    if not longest_special_path_info or len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    target_length, target_min_nodes = longest_special_path_info
    
    if target_length < 0 or target_min_nodes < 1:
        raise ValueError("Invalid input")
    
    n = len(edges) + 1
    
    # Build the tree
    graph = defaultdict(list)
    for u, v, length in edges:
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Find parent-child relationships by doing BFS from root 0
    parent = [-1] * n
    children = defaultdict(list)
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append(neighbor)
                queue.append(neighbor)
    
    # Handle the case where we need a path of length 0 (single node)
    if target_length == 0:
        nums = [0] * n
        return nums
    
    # Handle the case where we need a path of length 1 (two nodes)
    if target_length == 1:
        nums = [0] * n
        # We need at least one parent-child pair with unique values
        if n >= 2:
            nums[0] = 0
            nums[children[0][0]] = 1
        return nums
    
    # For longer paths, we need to construct a path of exactly target_length nodes
    # with all unique values
    nums = [0] * n
    
    # We'll create a chain from root downwards with unique values
    # The chain will have target_length nodes
    if target_length > n:
        raise ValueError("Invalid input")
    
    # Start from root and go down target_length nodes
    current = 0
    path_nodes = []
    for i in range(target_length):
        path_nodes.append(current)
        if i < target_length - 1:
            # Move to next node in the path
            if children[current]:
                current = children[current][0]
            else:
                # If we don't have enough nodes for the path, it's invalid
                raise ValueError("Invalid input")
    
    # Assign unique values to path nodes
    for i, node in enumerate(path_nodes):
        nums[node] = i
    
    # Fill remaining nodes with values that won't create longer paths
    # Use values that repeat to ensure no longer unique paths
    next_val = target_length
    for i in range(n):
        if nums[i] == 0 and i not in path_nodes:
            nums[i] = next_val % (target_length + 1)
            next_val += 1
    
    return nums
