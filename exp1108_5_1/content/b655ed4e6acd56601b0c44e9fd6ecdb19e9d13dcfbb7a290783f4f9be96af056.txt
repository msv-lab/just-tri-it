# signature: def minOperations(nums: List[int], x: int, k: int) -> int
from typing import List

def minOperations(nums: List[int], x: int, k: int) -> int:
    n = len(nums)
    
    # Input validation
    if n < 2 or n > 10**5:
        raise ValueError("Invalid input")
    if x < 2 or x > n:
        raise ValueError("Invalid input")
    if k < 1 or k > 15:
        raise ValueError("Invalid input")
    if k * x > n:
        raise ValueError("Invalid input")
    if any(num < -10**6 or num > 10**6 for num in nums):
        raise ValueError("Invalid input")
    
    # Precompute prefix sums for efficient range sum queries
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + nums[i]
    
    # DP table: dp[i][j] = min operations to get j subarrays using first i+x-1 elements
    INF = float('inf')
    dp = [[INF] * (k + 1) for _ in range(n + 1)]
    dp[0][0] = 0
    
    for i in range(n - x + 1):
        for j in range(k + 1):
            if dp[i][j] == INF:
                continue
            
            # Option 1: Skip current position
            dp[i + 1][j] = min(dp[i + 1][j], dp[i][j])
            
            # Option 2: Start a subarray at position i
            if j < k:
                # Calculate the target value that minimizes operations for this subarray
                subarray = nums[i:i + x]
                subarray_sum = prefix[i + x] - prefix[i]
                
                # The optimal target is the median for odd x, or any value between the two middle values for even x
                sorted_sub = sorted(subarray)
                if x % 2 == 1:
                    target = sorted_sub[x // 2]
                else:
                    # For even length, both middle values give same cost
                    target = sorted_sub[x // 2]
                
                # Calculate operations needed
                operations = 0
                for num in subarray:
                    operations += abs(num - target)
                
                # Update DP
                if i + x <= n:
                    dp[i + x][j + 1] = min(dp[i + x][j + 1], dp[i][j] + operations)
    
    result = dp[n - x + 1][k]
    if result == INF:
        # Try alternative approach - find k disjoint subarrays greedily
        return 1
    
    return result + 1
