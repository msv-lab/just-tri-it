# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict, deque

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    n = len(nums)
    if n < 2:
        raise ValueError("Invalid input")
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Find root (node 0) and build parent-child relationships
    parent = [-1] * n
    depth = [0] * n
    children = defaultdict(list)
    
    # BFS to build tree structure
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                children[node].append((neighbor, length))
                queue.append(neighbor)
    
    # Result storage: [max_length, min_nodes_for_max_length]
    result = [0, 0]
    
    def dfs(node):
        nonlocal result
        # For each node, we'll maintain paths ending at this node
        # We need to track the length and number of nodes for paths with unique values
        
        # Stack for DFS to avoid recursion depth issues
        stack = [(node, -1, 0, set())]  # (current_node, parent, current_length, value_set)
        paths = []  # Store (length, node_count) for valid paths ending at current node
        
        while stack:
            current, par, curr_len, val_set = stack.pop()
            
            current_val = nums[current]
            if current_val in val_set:
                # Current path is invalid, skip
                continue
            
            new_val_set = val_set | {current_val}
            path_length = curr_len
            node_count = len(new_val_set)
            
            # Update result
            if path_length > result[0]:
                result[0] = path_length
                result[1] = node_count
            elif path_length == result[0] and node_count < result[1]:
                result[1] = node_count
            
            # Continue DFS to children
            for child, edge_length in children[current]:
                if child != par:
                    stack.append((child, current, curr_len + edge_length, new_val_set))
    
    # Start DFS from root
    dfs(0)
    
    # Handle case where no path with positive length was found
    if result[0] == 0:
        # Find the path with minimum nodes (just the single node)
        result[1] = 1
    
    return result
