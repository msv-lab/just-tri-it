from typing import List
from collections import defaultdict, deque

def longestSpecialPath_inv_nums(longest_special_path_info: List[int], edges: List[List[int]]) -> List[int]:
    if not longest_special_path_info or len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    target_length, target_min_nodes = longest_special_path_info
    
    if target_length < 0 or target_min_nodes < 1:
        raise ValueError("Invalid input")
    
    n = len(edges) + 1
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Build tree structure with parent-child relationships (root at 0)
    children = defaultdict(list)
    parent = [-1] * n
    visited = [False] * n
    
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append((neighbor, length))
                queue.append(neighbor)
    
    # If target_length is 0, we need a path with 1 node (just a single node)
    if target_length == 0:
        if target_min_nodes != 1:
            raise ValueError("Invalid input")
        return [0] * n
    
    # For target_length >= 1, we need to create a path with target_min_nodes nodes
    # The path length in terms of edges is target_min_nodes - 1
    
    # Check if the requested path is possible
    if target_min_nodes < 1 or target_min_nodes > n:
        raise ValueError("Invalid input")
    
    # Strategy: Create a path from root with target_min_nodes nodes having unique values
    nums = [0] * n
    
    # Find a path from root with exactly target_min_nodes nodes
    path_nodes = []
    current = 0
    path_nodes.append(current)
    
    # Use DFS to find a path with exactly target_min_nodes nodes
    def dfs(node, path):
        if len(path) == target_min_nodes:
            return path[:]
        
        for child, length in children[node]:
            path.append(child)
            result = dfs(child, path)
            if result:
                return result
            path.pop()
        
        return None
    
    path = dfs(0, [0])
    
    if not path:
        # If we can't find a path with exactly target_min_nodes nodes from root,
        # we need to be more creative. Let's try to find any path with target_min_nodes nodes
        # by doing BFS from root to find the deepest path
        max_depth_nodes = []
        max_depth = 0
        depth = [0] * n
        queue = deque([0])
        
        while queue:
            node = queue.popleft()
            for child, length in children[node]:
                depth[child] = depth[node] + 1
                queue.append(child)
                if depth[child] > max_depth:
                    max_depth = depth[child]
                    max_depth_nodes = [child]
                elif depth[child] == max_depth:
                    max_depth_nodes.append(child)
        
        # Find the path to one of the deepest nodes
        if max_depth + 1 >= target_min_nodes:
            # Take one of the deepest nodes and backtrack to get the path
            end_node = max_depth_nodes[0]
            path = []
            current = end_node
            while current != -1:
                path.append(current)
                current = parent[current]
            path.reverse()
            # Take the first target_min_nodes nodes from this path
            path = path[:target_min_nodes]
        else:
            # We can't create a path with target_min_nodes nodes
            raise ValueError("Invalid input")
    
    # Assign unique values to the path nodes
    for i, node in enumerate(path):
        nums[node] = i
    
    # Fill remaining nodes with values that won't create longer special paths
    # Use values that are already used in the path to ensure uniqueness is maintained
    # only along our desired path
    used_values = set(nums)
    next_val = len(path)
    
    for i in range(n):
        if nums[i] == 0 and i not in path:
            # Assign a value that's already in the path to break potential longer paths
            nums[i] = 0
    
    return nums
