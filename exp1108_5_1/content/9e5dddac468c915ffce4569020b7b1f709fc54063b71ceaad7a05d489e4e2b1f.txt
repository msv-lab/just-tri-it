# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    n = len(nums)
    if n < 2 or len(edges) != n - 1:
        raise ValueError("Invalid input")
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # DFS to find parent-child relationships and build tree
    parent = [-1] * n
    depth = [0] * n
    dist_from_root = [0] * n
    
    stack = [0]
    parent[0] = -1
    depth[0] = 0
    dist_from_root[0] = 0
    
    while stack:
        node = stack.pop()
        for neighbor, length in graph[node]:
            if neighbor != parent[node]:
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                dist_from_root[neighbor] = dist_from_root[node] + length
                stack.append(neighbor)
    
    # Find all ancestor-descendant pairs with unique values
    max_length = 0
    min_nodes = float('inf')
    
    # Check all pairs of nodes where u is ancestor of v
    for u in range(n):
        for v in range(u, n):
            # Check if u is ancestor of v
            if u == v:
                # Single node path
                path_length = 0
                num_nodes = 1
                if path_length > max_length:
                    max_length = path_length
                    min_nodes = num_nodes
                elif path_length == max_length:
                    min_nodes = min(min_nodes, num_nodes)
                continue
            
            # Check if u is ancestor of v
            current = v
            is_ancestor = False
            while current != -1 and current != u:
                current = parent[current]
            if current == u:
                is_ancestor = True
            
            if is_ancestor:
                # Check if all values in path are unique
                current = v
                seen_values = set()
                all_unique = True
                path_nodes = []
                
                while current != -1:
                    if nums[current] in seen_values:
                        all_unique = False
                        break
                    seen_values.add(nums[current])
                    path_nodes.append(current)
                    if current == u:
                        break
                    current = parent[current]
                
                if all_unique:
                    path_length = abs(dist_from_root[v] - dist_from_root[u])
                    num_nodes = len(path_nodes)
                    
                    if path_length > max_length:
                        max_length = path_length
                        min_nodes = num_nodes
                    elif path_length == max_length:
                        min_nodes = min(min_nodes, num_nodes)
    
    return [max_length, min_nodes]
