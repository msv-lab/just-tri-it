# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
# test name: test_deep_alternating_duplicates
def test_deep_alternating_duplicates():
    """Test with alternating duplicate values in a deep chain"""
    edges = [[0,1,1],[1,2,1],[2,3,1],[3,4,1],[4,5,1],[5,6,1]]
    nums = [1,2,1,2,1,2,3]  # Alternating duplicates: 0&2&4 (1), 1&3&5 (2)
    result = longestSpecialPath(edges, nums)
    # Valid paths: 4->5->6 (length 2, 3 nodes), 2->3->4 (length 2, 3 nodes)
    # 0->1 (length 1, 2 nodes), 6 (length 0, 1 node)
    assert result == [2, 3]
