# signature: def postcondition(nums: List[int], x: int, k: int, min_operations_for_k_equal_subarrays: int) -> bool
from typing import List
import sys

def postcondition(nums: List[int], x: int, k: int, min_operations_for_k_equal_subarrays: int) -> bool:
    # Validate inputs
    if not (2 <= len(nums) <= 10**5):
        raise ValueError("Invalid input")
    if not all(-10**6 <= num <= 10**6 for num in nums):
        raise ValueError("Invalid input")
    if not (2 <= x <= len(nums)):
        raise ValueError("Invalid input")
    if not (1 <= k <= 15):
        raise ValueError("Invalid input")
    if not (2 <= k * x <= len(nums)):
        raise ValueError("Invalid input")
    if min_operations_for_k_equal_subarrays < 0:
        raise ValueError("Invalid input")
    
    n = len(nums)
    
    # Precompute prefix sums for efficient range sum calculation
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + nums[i]
    
    # DP state: dp[i][j] = minimum operations to get j valid subarrays using first i elements
    # Initialize with infinity
    INF = float('inf')
    dp = [[INF] * (k + 1) for _ in range(n + 1)]
    dp[0][0] = 0
    
    for i in range(1, n + 1):
        # Copy previous state (skip current element)
        for j in range(k + 1):
            dp[i][j] = min(dp[i][j], dp[i - 1][j])
        
        # Try to form a subarray ending at position i-1
        if i >= x:
            start_idx = i - x
            # Calculate the sum of the subarray from start_idx to i-1
            subarray_sum = prefix[i] - prefix[start_idx]
            
            # The optimal value to set all elements to is the median (for odd x) 
            # or average of two medians (for even x) to minimize operations
            # For simplicity and efficiency, we'll use the median approach
            
            # Extract the subarray
            subarray = nums[start_idx:i]
            sorted_sub = sorted(subarray)
            
            # For odd x, median is at position x//2
            # For even x, any value between the two middle elements gives same cost
            median = sorted_sub[x // 2]
            
            # Calculate cost to make all elements equal to median
            cost = 0
            for num in subarray:
                cost += abs(num - median)
            
            # Update DP state
            for j in range(1, k + 1):
                if dp[start_idx][j - 1] != INF:
                    dp[i][j] = min(dp[i][j], dp[start_idx][j - 1] + cost)
    
    # Find the minimum operations needed for at least k subarrays
    min_required = INF
    for i in range(n + 1):
        if dp[i][k] < min_required:
            min_required = dp[i][k]
    
    # If we couldn't find k subarrays, check if it's possible at all
    if min_required == INF:
        return False
    
    # The given value is valid if it's >= the minimum required operations
    return min_operations_for_k_equal_subarrays >= min_required
