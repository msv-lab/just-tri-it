# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
# test name: test_multiple_correct_outputs_scenario
def test_multiple_correct_outputs_scenario():
    """Test case where multiple valid longest paths exist with same characteristics"""
    edges = [[0,1,3],[0,2,3],[1,3,2],[2,4,2]]
    nums = [1,2,3,4,5]  # All unique values
    result = longestSpecialPath(edges, nums)
    # Both 0->1->3 and 0->2->4 have length 5 and 3 nodes
    # The function can return either, but both have same [5, 3]
    assert result[0] == 5 and result[1] == 3
