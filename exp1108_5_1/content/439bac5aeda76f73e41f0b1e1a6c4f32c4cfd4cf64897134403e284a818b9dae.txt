# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    n = len(nums)
    if n < 2 or len(edges) != n - 1:
        raise ValueError("Invalid input")
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Build tree structure with parent pointers
    parent = [-1] * n
    children = [[] for _ in range(n)]
    distances = [0] * n  # distance from root
    
    stack = [0]
    visited = [False] * n
    visited[0] = True
    
    while stack:
        node = stack.pop()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append((neighbor, length))
                distances[neighbor] = distances[node] + length
                stack.append(neighbor)
    
    max_length = 0
    min_nodes = float('inf')
    
    def dfs(node, path_values, path_lengths):
        nonlocal max_length, min_nodes
        
        current_value = nums[node]
        if current_value in path_values:
            return
        
        # Add current node to path
        path_values.add(current_value)
        
        # Calculate path length from root to current node
        current_path_length = distances[node]
        
        # Update global maximum
        if current_path_length > max_length:
            max_length = current_path_length
            min_nodes = len(path_values)
        elif current_path_length == max_length:
            min_nodes = min(min_nodes, len(path_values))
        
        # Continue DFS to children
        for child, length in children[node]:
            dfs(child, path_values.copy(), path_lengths)
    
    # Start DFS from root
    for i in range(n):
        dfs(i, set(), [])
    
    # Handle case where no path with positive length exists
    if max_length == 0:
        # All single-node paths have length 0
        return [0, 1]
    
    return [max_length, min_nodes]
