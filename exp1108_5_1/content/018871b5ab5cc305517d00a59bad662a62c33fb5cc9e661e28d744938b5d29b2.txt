# signature: def longestSpecialPath_inv_nums(longest_special_path_info: List[int], edges: List[List[int]]) -> List[int]
from typing import List
from collections import defaultdict, deque

def longestSpecialPath_inv_nums(longest_special_path_info: List[int], edges: List[List[int]]) -> List[int]:
    if not longest_special_path_info or len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    target_length, target_min_nodes = longest_special_path_info
    
    if target_length < 1 or target_min_nodes < 1 or target_min_nodes > target_length:
        raise ValueError("Invalid input")
    
    n = len(edges) + 1
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Build tree structure with parent-child relationships (root at 0)
    parent = [-1] * n
    children = defaultdict(list)
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append(neighbor)
                queue.append(neighbor)
    
    # Strategy: Create a path of length target_length with target_min_nodes unique values
    # We'll create a simple path from root downwards with the required properties
    
    nums = [0] * n
    
    if target_length == 1:
        # Single node path
        for i in range(n):
            nums[i] = i
        return nums
    
    # Find a path from root with exactly target_length nodes
    path_nodes = []
    
    def find_path(node, current_path):
        nonlocal path_nodes
        if path_nodes:
            return
        
        current_path.append(node)
        
        if len(current_path) == target_length:
            path_nodes = current_path.copy()
            return
        
        for child in children[node]:
            find_path(child, current_path)
            if path_nodes:
                return
        
        current_path.pop()
    
    find_path(0, [])
    
    if not path_nodes:
        # If we can't find a path of exact length, try to create one
        # Let's do BFS to find the longest path from root
        max_depth_nodes = []
        max_depth = 0
        
        stack = [(0, 0, -1)]  # (node, depth, parent)
        depths = [0] * n
        
        while stack:
            node, depth, par = stack.pop()
            depths[node] = depth
            
            if depth > max_depth:
                max_depth = depth
                max_depth_nodes = [node]
            elif depth == max_depth:
                max_depth_nodes.append(node)
            
            for child in children[node]:
                if child != par:
                    stack.append((child, depth + 1, node))
        
        # Take any node at max depth and reconstruct path to root
        if max_depth + 1 >= target_length:
            end_node = max_depth_nodes[0]
            path = []
            current = end_node
            while current != -1 and len(path) < target_length:
                path.append(current)
                current = parent[current]
            path.reverse()
            path_nodes = path[:target_length]
        else:
            # We need to create a longer path than exists in the tree
            # This is impossible, return a valid assignment anyway
            for i in range(n):
                nums[i] = i
            return nums
    
    # Assign values to create the special path
    # We need exactly target_min_nodes unique values in the path
    used_values = set()
    
    # Assign unique values to target_min_nodes nodes in the path
    unique_indices = []
    if target_min_nodes <= len(path_nodes):
        # Distribute unique values evenly along the path
        step = max(1, len(path_nodes) // target_min_nodes)
        for i in range(0, len(path_nodes), step):
            if len(unique_indices) < target_min_nodes:
                unique_indices.append(i)
    
    # Make sure we have exactly target_min_nodes unique indices
    while len(unique_indices) < target_min_nodes and len(unique_indices) < len(path_nodes):
        for i in range(len(path_nodes)):
            if i not in unique_indices:
                unique_indices.append(i)
                break
    
    # Assign values
    next_val = 0
    for i, node in enumerate(path_nodes):
        if i in unique_indices:
            # Assign unique value
            while next_val in used_values:
                next_val += 1
            nums[node] = next_val
            used_values.add(next_val)
            next_val += 1
        else:
            # Assign duplicate value (use value from first node in path)
            nums[node] = nums[path_nodes[0]]
    
    # Assign values to remaining nodes
    for i in range(n):
        if nums[i] == 0 and i not in path_nodes:
            # Use a value that doesn't conflict with used values
            val = 0
            while val in used_values:
                val += 1
            nums[i] = val
            used_values.add(val)
    
    return nums
