# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict, deque

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    n = len(nums)
    if n < 2:
        raise ValueError("Invalid input")
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Build tree structure with parent-child relationships (root at 0)
    parent = [-1] * n
    children = defaultdict(list)
    edge_lengths = defaultdict(dict)  # edge_lengths[u][v] = length
    
    # Store edge lengths
    for u, v, length in edges:
        edge_lengths[u][v] = length
        edge_lengths[v][u] = length
    
    # BFS to build tree structure
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append(neighbor)
                queue.append(neighbor)
    
    # Results: (max_length, min_nodes)
    result = [0, float('inf')]
    
    def dfs(node):
        nonlocal result
        
        # For each child, we'll maintain paths that end at current node
        # We need to track: path length, number of nodes, and the set of values
        
        # Base case: path starting and ending at current node
        current_paths = [(0, 1, {nums[node]})]  # (length, nodes, values)
        
        for child in children[node]:
            child_paths = dfs(child)
            
            # Add the edge from node to child to each child path
            edge_len = edge_lengths[node][child]
            for length, nodes_count, values in child_paths:
                if nums[node] not in values:
                    new_length = length + edge_len
                    new_nodes = nodes_count + 1
                    new_values = values | {nums[node]}
                    current_paths.append((new_length, new_nodes, new_values))
        
        # Update global result
        for length, nodes_count, _ in current_paths:
            if length > result[0]:
                result[0] = length
                result[1] = nodes_count
            elif length == result[0] and nodes_count < result[1]:
                result[1] = nodes_count
        
        return current_paths
    
    dfs(0)
    
    # Handle case where no path with positive length exists
    if result[0] == 0:
        result[1] = 1
    
    # Append 1 to the result as specified
    return [result[0], result[1], 1]
