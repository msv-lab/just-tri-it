# signature: def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool
from typing import List
from collections import defaultdict, deque

def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool:
    n = len(nums)
    
    # Validate input constraints
    if n < 2 or n > 50000:
        raise ValueError("Invalid input")
    if len(edges) != n - 1:
        raise ValueError("Invalid input")
    for edge in edges:
        if len(edge) != 3:
            raise ValueError("Invalid input")
        u, v, length = edge
        if u < 0 or u >= n or v < 0 or v >= n or length < 1 or length > 1000:
            raise ValueError("Invalid input")
    if len(nums) != n:
        raise ValueError("Invalid input")
    for num in nums:
        if num < 0 or num > 50000:
            raise ValueError("Invalid input")
    if len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    # Build tree adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Build parent-child relationships (tree rooted at 0)
    parent = [-1] * n
    depth = [0] * n
    dist_from_root = [0] * n
    
    # BFS to establish parent-child relationships and distances
    queue = deque([0])
    parent[0] = -1
    depth[0] = 0
    dist_from_root[0] = 0
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if neighbor != parent[node]:
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                dist_from_root[neighbor] = dist_from_root[node] + length
                queue.append(neighbor)
    
    # Find all special paths and track the longest ones
    max_path_length = 0
    min_nodes_for_max_length = float('inf')
    
    # For each node as the starting point of a path
    for start in range(n):
        # DFS to find all downward paths starting from 'start' with unique values
        stack = [(start, {nums[start]}, 0)]  # (current_node, values_set, path_length)
        
        while stack:
            node, values, path_length = stack.pop()
            
            # Update maximum path length and minimum nodes
            if path_length > max_path_length:
                max_path_length = path_length
                min_nodes_for_max_length = len(values)
            elif path_length == max_path_length:
                min_nodes_for_max_length = min(min_nodes_for_max_length, len(values))
            
            # Explore children
            for neighbor, edge_length in graph[node]:
                if neighbor != parent[node]:  # Only go downward
                    if nums[neighbor] not in values:  # Check if value is unique
                        new_values = values.copy()
                        new_values.add(nums[neighbor])
                        new_path_length = path_length + edge_length
                        stack.append((neighbor, new_values, new_path_length))
    
    # Check if the claimed info matches our computed values
    claimed_length, claimed_min_nodes = longest_special_path_info
    
    # Allow for floating point precision issues in path length comparison
    if abs(claimed_length - max_path_length) > 1e-9:
        return False
    
    if claimed_min_nodes != min_nodes_for_max_length:
        return False
    
    return True
