# signature: def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool
from typing import List
from collections import defaultdict, deque

def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool:
    n = len(nums)
    
    # Validate input constraints
    if n < 2 or n > 50000:
        raise ValueError("Invalid input")
    if len(edges) != n - 1:
        raise ValueError("Invalid input")
    if len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    # Build adjacency list for the tree
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length < 1 or length > 1000:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Build the tree structure with parent-child relationships (root at 0)
    parent = [-1] * n
    children = [[] for _ in range(n)]
    distances = [0] * n  # distance from root
    
    # BFS to build tree structure and calculate distances from root
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append((neighbor, length))
                distances[neighbor] = distances[node] + length
                queue.append(neighbor)
    
    # Find all special paths and track the longest ones
    max_path_length = 0
    min_nodes_for_max_length = float('inf')
    
    # Helper function for DFS to find special paths
    def dfs(node, current_path_values, current_path_nodes):
        nonlocal max_path_length, min_nodes_for_max_length
        
        # Current path is special (all values are unique)
        # Calculate path length from root to current node minus root to LCA
        if len(current_path_nodes) > 0:
            # For any path from ancestor to descendant, the path length is distance[descendant] - distance[ancestor]
            # We need to check all subpaths in current_path_nodes
            for i in range(len(current_path_nodes)):
                for j in range(i, len(current_path_nodes)):
                    ancestor = current_path_nodes[i]
                    descendant = current_path_nodes[j]
                    path_length = distances[descendant] - distances[ancestor]
                    
                    if path_length > max_path_length:
                        max_path_length = path_length
                        min_nodes_for_max_length = j - i + 1
                    elif path_length == max_path_length:
                        min_nodes_for_max_length = min(min_nodes_for_max_length, j - i + 1)
        
        # Explore children
        for child, length in children[node]:
            if nums[child] not in current_path_values:
                current_path_values.add(nums[child])
                current_path_nodes.append(child)
                dfs(child, current_path_values, current_path_nodes)
                current_path_nodes.pop()
                current_path_values.remove(nums[child])
    
    # Start DFS from root
    dfs(0, {nums[0]}, [0])
    
    # Check if the claimed values match the actual values
    claimed_length = longest_special_path_info[0]
    claimed_min_nodes = longest_special_path_info[1]
    
    return claimed_length == max_path_length and claimed_min_nodes == min_nodes_for_max_length
