# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    n = len(nums)
    
    # Validate input
    if n < 2 or len(edges) != n - 1:
        raise ValueError("Invalid input")
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Find root (node 0) and build parent-child relationships
    parent = [-1] * n
    children = defaultdict(list)
    edge_length = {}
    
    stack = [0]
    visited = [False] * n
    visited[0] = True
    
    while stack:
        node = stack.pop()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append(neighbor)
                edge_length[(node, neighbor)] = length
                stack.append(neighbor)
    
    # Results: [max_length, min_nodes_for_max_length]
    result = [0, 0]
    
    def dfs(node):
        nonlocal result
        
        # Store paths ending at current node: {node_value: (length, node_count)}
        # We'll maintain the best path for each unique value in the current subtree
        paths_from_node = {}
        
        # Initialize with the current node itself
        paths_from_node[nums[node]] = (0, 1)  # (length, node_count)
        
        for child in children[node]:
            child_paths = dfs(child)
            edge_len = edge_length[(node, child)]
            
            # Try to extend paths from child to current node
            for val, (child_length, child_nodes) in child_paths.items():
                if val != nums[node]:  # Check if we can extend without duplicate value
                    new_length = child_length + edge_len
                    new_nodes = child_nodes + 1
                    
                    # Update result if we found a better path
                    if new_length > result[0]:
                        result[0] = new_length
                        result[1] = new_nodes
                    elif new_length == result[0] and new_nodes < result[1]:
                        result[1] = new_nodes
                    
                    # Update paths_from_node for the current node
                    if val not in paths_from_node or new_length > paths_from_node[val][0]:
                        paths_from_node[val] = (new_length, new_nodes)
        
        # Also check paths starting and ending at current node
        if 0 > result[0]:
            result[0] = 0
            result[1] = 1
        elif 0 == result[0] and 1 < result[1]:
            result[1] = 1
        
        return paths_from_node
    
    dfs(0)
    
    # Handle case where no path was found (shouldn't happen but just in case)
    if result[0] == 0 and result[1] == 0:
        result[1] = 1
    
    return result
