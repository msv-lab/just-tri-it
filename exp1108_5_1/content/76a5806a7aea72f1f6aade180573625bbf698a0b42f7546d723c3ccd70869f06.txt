# signature: def minOperations_sinv_k(min_operations_for_k_equal_subarrays: int, nums: List[int], x: int) -> list[int]
from typing import List
import math

def minOperations_sinv_k(min_operations_for_k_equal_subarrays: int, nums: List[int], x: int) -> List[int]:
    n = len(nums)
    
    # Validate inputs
    if n < 2 or n > 10**5:
        raise ValueError("Invalid input")
    if x < 2 or x > n:
        raise ValueError("Invalid input")
    if any(num < -10**6 or num > 10**6 for num in nums):
        raise ValueError("Invalid input")
    if min_operations_for_k_equal_subarrays < 0:
        raise ValueError("Invalid input")
    
    # Precompute prefix sums for efficient range sum calculation
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + nums[i]
    
    # For each possible starting position of a subarray of length x,
    # calculate the cost to make all elements equal to their median
    # Using median minimizes the sum of absolute differences
    
    subarray_costs = []
    for i in range(n - x + 1):
        # Extract the subarray
        subarr = nums[i:i + x]
        
        # Sort to find median
        sorted_subarr = sorted(subarr)
        
        # For odd length, median is middle element
        # For even length, either middle element works (both give same cost)
        median = sorted_subarr[x // 2]
        
        # Calculate cost to make all elements equal to median
        cost = 0
        for num in subarr:
            cost += abs(num - median)
        
        subarray_costs.append((i, cost))
    
    # Now we need to find k non-overlapping subarrays with minimum total cost
    # and check if that minimum cost equals the given target
    
    result = []
    max_possible_k = n // x
    
    # Check all possible k values within constraints
    for k in range(1, min(16, max_possible_k + 1)):
        if k * x > n:
            continue
            
        # Use dynamic programming to find minimum operations for k subarrays
        # dp[i][j] = min cost to choose j subarrays from first i positions
        # But with constraints, we need to be careful about overlaps
        
        # Since k is small (<=15), we can use a DP that considers positions
        dp = [[float('inf')] * (k + 1) for _ in range(n + 1)]
        dp[0][0] = 0
        
        for i in range(n + 1):
            for j in range(k + 1):
                if dp[i][j] == float('inf'):
                    continue
                
                # Option 1: Skip current position
                if i + 1 <= n:
                    dp[i + 1][j] = min(dp[i + 1][j], dp[i][j])
                
                # Option 2: Start a subarray at position i (if valid)
                if j < k and i + x <= n:
                    start_idx = i
                    cost = subarray_costs[start_idx][1] if start_idx < len(subarray_costs) else float('inf')
                    dp[i + x][j + 1] = min(dp[i + x][j + 1], dp[i][j] + cost)
        
        # Find minimum cost for exactly k subarrays
        min_cost = float('inf')
        for i in range(n + 1):
            min_cost = min(min_cost, dp[i][k])
        
        if min_cost == min_operations_for_k_equal_subarrays:
            result.append(k)
    
    return sorted(result)
