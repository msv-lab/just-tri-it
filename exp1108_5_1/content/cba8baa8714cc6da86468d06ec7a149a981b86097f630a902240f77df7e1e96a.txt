# signature: def postcondition(nums: List[int], x: int, k: int, min_operations_for_k_equal_subarrays: int) -> bool
from typing import List

def postcondition(nums: List[int], x: int, k: int, min_operations_for_k_equal_subarrays: int) -> bool:
    n = len(nums)
    
    # Validate input constraints
    if n < 2 or n > 10**5:
        raise ValueError("Invalid input")
    if any(num < -10**6 or num > 10**6 for num in nums):
        raise ValueError("Invalid input")
    if x < 2 or x > n:
        raise ValueError("Invalid input")
    if k < 1 or k > 15:
        raise ValueError("Invalid input")
    if k * x < 2 or k * x > n:
        raise ValueError("Invalid input")
    if min_operations_for_k_equal_subarrays < 0:
        raise ValueError("Invalid input")
    
    # Precompute prefix sums for efficient range sum calculation
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + nums[i]
    
    # DP state: dp[i][j] = minimum operations to get j valid subarrays using first i elements
    # We use a large number as infinity
    INF = 10**18
    dp = [[INF] * (k + 1) for _ in range(n + 1)]
    dp[0][0] = 0
    
    for i in range(n + 1):
        for j in range(k + 1):
            if dp[i][j] == INF:
                continue
            
            # Option 1: Skip current element (don't start a new subarray)
            if i < n:
                dp[i + 1][j] = min(dp[i + 1][j], dp[i][j])
            
            # Option 2: Start a new subarray of length x at position i
            if j < k and i + x <= n:
                # Calculate the cost to make all elements in subarray [i, i+x-1] equal
                # The optimal target value is the median (for odd x) or average of two medians (for even x)
                # But for simplicity and efficiency, we'll use the approach of making all equal to one value
                # that minimizes the sum of absolute differences
                
                # Extract the subarray
                subarray = nums[i:i + x]
                
                # Sort to find median
                sorted_sub = sorted(subarray)
                
                # For odd length, median is middle element
                # For even length, any value between the two middle elements gives same cost
                if x % 2 == 1:
                    median = sorted_sub[x // 2]
                else:
                    # For even length, we can use either middle element
                    median = sorted_sub[x // 2]
                
                # Calculate cost to make all elements equal to median
                cost = 0
                for num in subarray:
                    cost += abs(num - median)
                
                # Alternative: we can also try making all elements equal to the mean (rounded)
                # to handle cases where mean might be better
                mean_val = sum(subarray) // x
                cost_mean = 0
                for num in subarray:
                    cost_mean += abs(num - mean_val)
                
                # Try mean_val + 1 as well (for better rounding)
                cost_mean_plus = 0
                for num in subarray:
                    cost_mean_plus += abs(num - (mean_val + 1))
                
                cost = min(cost, cost_mean, cost_mean_plus)
                
                # Update DP state
                dp[i + x][j + 1] = min(dp[i + x][j + 1], dp[i][j] + cost)
    
    # Find the minimum operations needed to get at least k subarrays
    min_required = INF
    for i in range(n + 1):
        min_required = min(min_required, dp[i][k])
    
    # The given value is valid if it's >= the minimum required operations
    return min_operations_for_k_equal_subarrays >= min_required and min_required != INF
