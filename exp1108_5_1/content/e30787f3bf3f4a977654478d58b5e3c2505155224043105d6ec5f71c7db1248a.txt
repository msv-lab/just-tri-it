# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict, deque

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    n = len(nums)
    if n < 2 or len(edges) != n - 1:
        raise ValueError("Invalid input")
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Build tree structure with parent pointers
    parent = [-1] * n
    children = defaultdict(list)
    edge_length = defaultdict(dict)
    
    # BFS to build tree rooted at 0
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append(neighbor)
                edge_length[node][neighbor] = length
                queue.append(neighbor)
    
    # DFS to compute longest special paths
    max_length = 0
    min_nodes = float('inf')
    
    def dfs(node, path_values, path_length, node_count):
        nonlocal max_length, min_nodes
        
        current_value = nums[node]
        if current_value in path_values:
            # Can't extend path with duplicate value
            return
        
        # Update path_values and path_length for current node
        new_path_values = path_values | {current_value}
        
        # Check if this path is valid (all unique values)
        if len(new_path_values) == node_count + 1:
            # Update global maximum
            if path_length > max_length:
                max_length = path_length
                min_nodes = node_count + 1
            elif path_length == max_length:
                min_nodes = min(min_nodes, node_count + 1)
        
        # Recursively process children
        for child in children[node]:
            edge_len = edge_length[node][child]
            dfs(child, new_path_values, path_length + edge_len, node_count + 1)
    
    # Start DFS from each node as potential starting point
    for start_node in range(n):
        dfs(start_node, set(), 0, 0)
    
    # Handle case where no path with more than 1 node is found
    if max_length == 0:
        min_nodes = 1
    
    result = [max_length, min_nodes]
    
    # Append 1 to the final answer as requested
    result.append(1)
    return result
