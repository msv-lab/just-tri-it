# signature: def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool
from typing import List
from collections import defaultdict, deque

def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool:
    n = len(nums)
    
    # Validate input constraints
    if n < 2 or n > 50000:
        raise ValueError("Invalid input")
    if len(edges) != n - 1:
        raise ValueError("Invalid input")
    if len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length < 1 or length > 1000:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Validate nums array
    if len(nums) != n:
        raise ValueError("Invalid input")
    for num in nums:
        if num < 0 or num > 50000:
            raise ValueError("Invalid input")
    
    # Build tree structure with parent-child relationships
    parent = [-1] * n
    children = defaultdict(list)
    edge_lengths = defaultdict(int)
    
    # BFS to build tree structure starting from root 0
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append(neighbor)
                edge_lengths[(node, neighbor)] = length
                queue.append(neighbor)
    
    # DFS to find all special paths
    max_path_length = 0
    min_nodes_for_max_length = float('inf')
    
    def dfs(node, current_path_values, current_path_length, current_nodes_count):
        nonlocal max_path_length, min_nodes_for_max_length
        
        # Current node value
        current_value = nums[node]
        
        # Check if current value is already in path
        if current_value in current_path_values:
            return
        
        # Add current node to path
        current_path_values.add(current_value)
        current_nodes_count += 1
        
        # Update global maximum
        if current_path_length > max_path_length:
            max_path_length = current_path_length
            min_nodes_for_max_length = current_nodes_count
        elif current_path_length == max_path_length:
            if current_nodes_count < min_nodes_for_max_length:
                min_nodes_for_max_length = current_nodes_count
        
        # Continue DFS to children
        for child in children[node]:
            edge_len = edge_lengths[(node, child)]
            dfs(child, current_path_values.copy(), current_path_length + edge_len, current_nodes_count)
    
    # Start DFS from each node as potential starting point
    for start_node in range(n):
        dfs(start_node, set(), 0, 0)
    
    # Handle case where no special path was found (shouldn't happen as single node paths are valid)
    if max_path_length == 0:
        max_path_length = 0
        min_nodes_for_max_length = 1
    
    # Check if the provided info matches our computed values
    claimed_length = longest_special_path_info[0]
    claimed_min_nodes = longest_special_path_info[1]
    
    return claimed_length == max_path_length and claimed_min_nodes == min_nodes_for_max_length
