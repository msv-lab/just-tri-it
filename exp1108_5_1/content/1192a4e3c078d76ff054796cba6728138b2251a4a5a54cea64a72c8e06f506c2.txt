# signature: def minOperations(nums: List[int], x: int, k: int) -> int
from typing import List

def minOperations(nums: List[int], x: int, k: int) -> int:
    n = len(nums)
    
    # Validate constraints
    if n < 2 or n > 10**5:
        raise ValueError("Invalid input")
    if x < 2 or x > n:
        raise ValueError("Invalid input")
    if k < 1 or k > 15:
        raise ValueError("Invalid input")
    if k * x > n:
        raise ValueError("Invalid input")
    
    # Precompute prefix sums for efficient range sum calculation
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + nums[i]
    
    # DP table: dp[i][j] = min operations to get j subarrays using first i+x-1 elements
    INF = float('inf')
    dp = [[INF] * (k + 1) for _ in range(n + 1)]
    
    # Base case: 0 operations needed for 0 subarrays
    for i in range(n + 1):
        dp[i][0] = 0
    
    # For each possible starting position of a subarray
    for start in range(n - x + 1):
        end = start + x - 1
        
        # Calculate cost to make all elements in this subarray equal to target value
        # The optimal target is the median of the subarray (for minimum absolute deviations)
        subarray = nums[start:end + 1]
        sorted_sub = sorted(subarray)
        
        # For odd length, median is middle element; for even, either middle works
        median = sorted_sub[x // 2]
        
        # Calculate operations needed
        cost = 0
        for num in subarray:
            cost += abs(num - median)
        
        # Update DP table
        for j in range(k, 0, -1):
            for i in range(end + 1, n + 1):
                if dp[start][j - 1] != INF:
                    dp[i][j] = min(dp[i][j], dp[start][j - 1] + cost)
    
    # Find the minimum operations to get at least k subarrays
    result = INF
    for i in range(n + 1):
        if dp[i][k] < result:
            result = dp[i][k]
    
    if result == INF:
        raise ValueError("Invalid input")
    
    return int(result)
