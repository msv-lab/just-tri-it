from typing import List
from collections import defaultdict, deque

def longestSpecialPath_inv_nums(longest_special_path_info: List[int], edges: List[List[int]]) -> List[int]:
    if len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    target_length, target_min_nodes = longest_special_path_info
    
    if target_length < 1 or target_min_nodes < 1 or target_min_nodes > target_length:
        raise ValueError("Invalid input")
    
    n = len(edges) + 1
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Build tree structure (rooted at 0)
    parent = [-1] * n
    children = defaultdict(list)
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append((neighbor, length))
                queue.append(neighbor)
    
    # Strategy: Create a path of length target_length with target_min_nodes
    # We'll create a chain where the first target_min_nodes nodes have unique values
    # and the remaining nodes repeat values to break the path
    
    nums = [0] * n
    
    if target_length == 1:
        # All nodes can have the same value
        return [0] * n
    
    # Find the longest chain from root
    chain_nodes = []
    stack = [(0, 0)]  # (node, depth)
    max_depth = 0
    deepest_node = 0
    
    while stack:
        node, depth = stack.pop()
        if depth > max_depth:
            max_depth = depth
            deepest_node = node
        for child, _ in children[node]:
            stack.append((child, depth + 1))
    
    # Build the path from root to deepest node
    path = []
    current = deepest_node
    while current != -1:
        path.append(current)
        current = parent[current]
    path.reverse()
    
    if len(path) < target_length:
        # Need to extend the path
        # Find another branch to extend
        extended_path = path[:]
        current = path[-1]
        
        # Try to find a child not in the current path
        for child, _ in children[current]:
            if child not in extended_path:
                # Extend the path with this child
                temp_path = extended_path + [child]
                if len(temp_path) >= target_length:
                    path = temp_path[:target_length]
                    break
                else:
                    # Do DFS to extend further
                    stack = [(child, len(temp_path))]
                    while stack and len(temp_path) < target_length:
                        node, depth = stack.pop()
                        temp_path.append(node)
                        if len(temp_path) == target_length:
                            break
                        for grandchild, _ in children[node]:
                            stack.append((grandchild, depth + 1))
                    if len(temp_path) >= target_length:
                        path = temp_path[:target_length]
                        break
        
        if len(path) < target_length:
            # If we still can't get the required length, create a linear chain
            # by using a DFS from root
            path = []
            stack = [(0, [0])]
            while stack and len(path) < target_length:
                node, current_path = stack.pop()
                if len(current_path) > len(path):
                    path = current_path[:]
                if len(path) >= target_length:
                    path = path[:target_length]
                    break
                for child, _ in children[node]:
                    if child not in current_path:
                        stack.append((child, current_path + [child]))
    
    # Assign values to create the special path
    # The first target_min_nodes nodes in the path get unique values
    # The remaining nodes get repeating values to break uniqueness
    
    if len(path) < target_length:
        raise ValueError("Invalid input: Cannot create path of required length")
    
    # Assign unique values to the first target_min_nodes nodes
    used_values = set()
    for i in range(target_min_nodes):
        node = path[i]
        # Assign a unique value not used before
        val = i
        while val in used_values:
            val += 1
        nums[node] = val
        used_values.add(val)
    
    # For the remaining nodes in the path, assign values that repeat
    # to ensure the path length is exactly target_length
    for i in range(target_min_nodes, target_length):
        node = path[i]
        # Assign a value that already exists in the path to break uniqueness
        nums[node] = nums[path[i - target_min_nodes]]
    
    # Fill remaining nodes with values that don't create longer paths
    for i in range(n):
        if nums[i] == 0 and i not in path:
            nums[i] = 0
    
    return nums
