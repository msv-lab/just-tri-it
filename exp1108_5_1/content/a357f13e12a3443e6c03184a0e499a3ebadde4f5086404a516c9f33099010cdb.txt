# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    n = len(nums)
    if n < 2 or len(edges) != n - 1:
        raise ValueError("Invalid input")
    
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    parent = [-1] * n
    depth = [0] * n
    dist_from_root = [0] * n
    
    stack = [0]
    parent[0] = 0
    while stack:
        node = stack.pop()
        for neighbor, length in graph[node]:
            if parent[neighbor] == -1:
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                dist_from_root[neighbor] = dist_from_root[node] + length
                stack.append(neighbor)
    
    result = [0, float('inf')]
    
    def dfs(node):
        nonlocal result
        path_info = [(0, 1, node)]
        seen_values = {nums[node]: node}
        
        for child, length in graph[node]:
            if child == parent[node]:
                continue
            child_info = dfs(child)
            for dist, cnt, end in child_info:
                new_dist = dist + length
                path_info.append((new_dist, cnt + 1, end))
                
                if nums[end] in seen_values:
                    continue
                seen_values[nums[end]] = end
                
                if new_dist > result[0]:
                    result[0] = new_dist
                    result[1] = cnt + 1
                elif new_dist == result[0]:
                    result[1] = min(result[1], cnt + 1)
        
        return path_info
    
    dfs(0)
    
    if result[0] == 0:
        result[1] = 1
    
    result[0] += 1
    result[1] += 1
    return result
