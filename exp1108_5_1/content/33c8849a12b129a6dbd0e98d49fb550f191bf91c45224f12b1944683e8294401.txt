# signature: def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool
from typing import List
from collections import defaultdict, deque

def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool:
    n = len(nums)
    
    # Validate input constraints
    if n < 2 or n > 50000:
        raise ValueError("Invalid input")
    
    if len(edges) != n - 1:
        raise ValueError("Invalid input")
    
    for edge in edges:
        if len(edge) != 3:
            raise ValueError("Invalid input")
        u, v, length = edge
        if u < 0 or u >= n or v < 0 or v >= n or length < 1 or length > 1000:
            raise ValueError("Invalid input")
    
    if len(nums) != n:
        raise ValueError("Invalid input")
    
    for num in nums:
        if num < 0 or num > 50000:
            raise ValueError("Invalid input")
    
    if len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    claimed_length, claimed_min_nodes = longest_special_path_info
    if claimed_length < 0 or claimed_min_nodes < 1:
        raise ValueError("Invalid input")
    
    # Build the tree
    graph = defaultdict(list)
    for u, v, length in edges:
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Build parent-child relationships (root at 0)
    parent = [-1] * n
    depth = [0] * n
    dist_from_root = [0] * n
    
    # BFS to establish parent-child relationships and distances
    queue = deque([0])
    visited = [False] * n
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                dist_from_root[neighbor] = dist_from_root[node] + length
                queue.append(neighbor)
    
    # Find all special paths and track the longest ones
    max_path_length = 0
    min_nodes_for_max_length = float('inf')
    
    # DFS to find special paths
    def dfs(node, path_values, path_nodes):
        nonlocal max_path_length, min_nodes_for_max_length
        
        current_value = nums[node]
        
        # Check if current value already exists in path
        if current_value in path_values:
            return
        
        # Add current node to path
        path_values.add(current_value)
        path_nodes.append(node)
        
        # Current path is special, check its length
        if len(path_nodes) >= 2:
            # Calculate path length (distance between first and last node in path)
            start_node = path_nodes[0]
            end_node = path_nodes[-1]
            
            # Calculate distance using LCA
            path_length = dist_from_root[start_node] + dist_from_root[end_node] - 2 * dist_from_root[find_lca(start_node, end_node)]
            
            if path_length > max_path_length:
                max_path_length = path_length
                min_nodes_for_max_length = len(path_nodes)
            elif path_length == max_path_length:
                min_nodes_for_max_length = min(min_nodes_for_max_length, len(path_nodes))
        
        # Continue DFS to children
        for neighbor, _ in graph[node]:
            if neighbor != parent[node]:
                dfs(neighbor, path_values.copy(), path_nodes.copy())
    
    # Preprocess for LCA
    LOG = (n).bit_length()
    up = [[-1] * n for _ in range(LOG)]
    
    # Initialize first level
    for i in range(n):
        up[0][i] = parent[i]
    
    # Build binary lifting table
    for i in range(1, LOG):
        for j in range(n):
            if up[i-1][j] != -1:
                up[i][j] = up[i-1][up[i-1][j]]
    
    def find_lca(a, b):
        if depth[a] < depth[b]:
            a, b = b, a
        
        # Lift a to same depth as b
        diff = depth[a] - depth[b]
        for i in range(LOG):
            if diff & (1 << i):
                a = up[i][a]
        
        if a == b:
            return a
        
        # Lift both until their parents are the same
        for i in range(LOG-1, -1, -1):
            if up[i][a] != up[i][b]:
                a = up[i][a]
                b = up[i][b]
        
        return up[0][a]
    
    # Start DFS from each node as potential starting point
    for i in range(n):
        dfs(i, set(), [])
    
    # Handle case where no special path with at least 2 nodes exists
    # A single node is considered a special path with length 0
    if max_path_length == 0:
        # Check if claimed info matches single node case
        return claimed_length == 0 and claimed_min_nodes == 1
    
    # Check if claimed info matches actual results
    return claimed_length == max_path_length and claimed_min_nodes == min_nodes_for_max_length
