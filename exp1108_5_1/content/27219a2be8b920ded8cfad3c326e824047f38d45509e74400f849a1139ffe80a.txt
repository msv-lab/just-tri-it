# signature: def postcondition(nums: List[int], x: int, k: int, min_operations_for_k_equal_subarrays: int) -> bool
from typing import List

def postcondition(nums: List[int], x: int, k: int, min_operations_for_k_equal_subarrays: int) -> bool:
    n = len(nums)
    
    # Validate constraints
    if n < 2 or n > 10**5:
        raise ValueError("Invalid input")
    if any(num < -10**6 or num > 10**6 for num in nums):
        raise ValueError("Invalid input")
    if x < 2 or x > n:
        raise ValueError("Invalid input")
    if k < 1 or k > 15:
        raise ValueError("Invalid input")
    if k * x < 2 or k * x > n:
        raise ValueError("Invalid input")
    if min_operations_for_k_equal_subarrays < 0:
        raise ValueError("Invalid input")
    
    # Precompute prefix sums for efficient range sum calculation
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + nums[i]
    
    # Memoization for DP
    memo = {}
    
    def dp(idx, remaining_k, last_end):
        if remaining_k == 0:
            return 0
        if idx > n - x:
            return float('inf')
        
        key = (idx, remaining_k, last_end)
        if key in memo:
            return memo[key]
        
        # Option 1: Skip current position
        skip_cost = dp(idx + 1, remaining_k, last_end)
        
        # Option 2: Use current position as start of a subarray
        if idx >= last_end:
            # Calculate cost to make all elements in subarray [idx, idx+x-1] equal
            # The optimal value to set all elements to is the median of the subarray
            subarray = nums[idx:idx + x]
            sorted_sub = sorted(subarray)
            median = sorted_sub[x // 2]
            
            # Calculate operations needed
            operations = 0
            for num in subarray:
                operations += abs(num - median)
            
            take_cost = operations + dp(idx + x, remaining_k - 1, idx + x)
        else:
            take_cost = float('inf')
        
        result = min(skip_cost, take_cost)
        memo[key] = result
        return result
    
    # Find minimum operations using DP
    min_required = dp(0, k, 0)
    
    # Check if the given value is valid
    return min_operations_for_k_equal_subarrays >= min_required
