from typing import List
from collections import defaultdict, deque

def longestSpecialPath_inv_nums(longest_special_path_info: List[int], edges: List[List[int]]) -> List[int]:
    if not longest_special_path_info or len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    target_length, target_min_nodes = longest_special_path_info
    
    if target_length < 0 or target_min_nodes < 1:
        raise ValueError("Invalid input")
    
    n = len(edges) + 1
    
    # Build the tree
    graph = defaultdict(list)
    for u, v, length in edges:
        graph[u].append(v)
        graph[v].append(u)
    
    # Find parent-child relationships using BFS from root 0
    parent = [-1] * n
    children = defaultdict(list)
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append(neighbor)
                queue.append(neighbor)
    
    # Case 1: If target_length is 0, we need a path with 1 node (just the node itself)
    if target_length == 0:
        if target_min_nodes != 1:
            raise ValueError("Invalid input")
        # All nodes can have the same value (e.g., 0) since no path longer than 1 node exists
        return [0] * n
    
    # Case 2: We need to create a path of length target_length with target_min_nodes nodes
    # The path length in terms of edges is target_length, number of nodes is target_length + 1
    if target_min_nodes != target_length + 1:
        raise ValueError("Invalid input")
    
    # We'll create a simple chain from root downwards with unique values
    nums = [0] * n
    
    # Start from root and create a chain of target_min_nodes nodes with unique values
    current = 0
    value = 0
    count = 0
    
    while current != -1 and count < target_min_nodes:
        nums[current] = value
        value += 1
        count += 1
        
        # Move to the first child if available
        if children[current]:
            current = children[current][0]
        else:
            break
    
    # Fill remaining nodes with values that will break uniqueness in any longer path
    remaining_value = 50000  # Use a large value that won't conflict
    
    def fill_remaining(node, visited_set):
        nonlocal remaining_value
        if nums[node] != 0:  # Already assigned
            return
        
        # Assign a value that repeats to break path uniqueness
        nums[node] = remaining_value
        remaining_value -= 1
        
        for child in children[node]:
            fill_remaining(child, visited_set)
    
    # Fill all unassigned nodes
    for i in range(n):
        if nums[i] == 0:
            fill_remaining(i, set())
    
    return nums
