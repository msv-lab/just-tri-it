def test_complex_star_with_various_duplicates():
    """Test star structure with mixed duplicate patterns"""
    edges = [[0,1,4],[0,2,3],[0,3,5],[0,4,2],[0,5,6]]
    nums = [1,2,1,3,2,4]  # Duplicates: 0&2 (1), 1&4 (2)
    result = longestSpecialPath(edges, nums)
    # Valid paths: 0->3 (length 5, 2 nodes), 0->5 (length 6, 2 nodes)
    # 0->1 (length 4, 2 nodes), 0->2 (length 3, 2 nodes), 0->4 (length 2, 2 nodes)
    assert result == [6, 2]
