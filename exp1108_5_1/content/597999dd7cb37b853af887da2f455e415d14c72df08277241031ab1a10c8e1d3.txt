# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict, deque

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    if not edges or len(nums) != len(edges) + 1:
        raise ValueError("Invalid input")
    
    n = len(nums)
    graph = defaultdict(list)
    
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Build tree structure with parent pointers
    parent = [-1] * n
    depth = [0] * n
    dist_from_root = [0] * n
    
    # BFS to build tree and calculate distances from root
    queue = deque([0])
    visited = [False] * n
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                dist_from_root[neighbor] = dist_from_root[node] + length
                queue.append(neighbor)
    
    # Find all paths with unique values using DFS
    max_length = 0
    min_nodes = float('inf')
    
    def dfs(node, path_values, path_nodes, current_length):
        nonlocal max_length, min_nodes
        
        # Check if current path is valid (all unique values)
        if len(path_values) == len(set(path_values)):
            if current_length > max_length:
                max_length = current_length
                min_nodes = len(path_nodes)
            elif current_length == max_length:
                min_nodes = min(min_nodes, len(path_nodes))
        
        # Explore children
        for neighbor, length in graph[node]:
            if neighbor != parent[node]:  # Only go downward
                # Add the neighbor to the path
                path_values.append(nums[neighbor])
                path_nodes.append(neighbor)
                
                dfs(neighbor, path_values, path_nodes, current_length + length)
                
                # Backtrack
                path_values.pop()
                path_nodes.pop()
    
    # Try starting from each node
    for start_node in range(n):
        dfs(start_node, [nums[start_node]], [start_node], 0)
    
    # Handle case where no valid path found (shouldn't happen due to single node paths)
    if max_length == 0:
        min_nodes = 1
    
    result = [max_length, min_nodes]
    
    # Append 1 to the result as specified
    result.append(1)
    return result
