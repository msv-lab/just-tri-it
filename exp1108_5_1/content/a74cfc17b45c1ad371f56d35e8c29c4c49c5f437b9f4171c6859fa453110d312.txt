# signature: def minCost_simp(cost: List[List[int]]) -> int
from typing import List

def minCost_simp(cost: List[List[int]]) -> int:
    n = len(cost)
    if n % 2 != 0 or n < 2 or n > 10**5:
        raise ValueError("Invalid input")
    
    for row in cost:
        if len(row) != 3:
            raise ValueError("Invalid input")
        for val in row:
            if not (0 <= val <= 10**5):
                raise ValueError("Invalid input")
    
    # For each pair of symmetric positions (i, n-1-i), we need to choose different colors
    # We'll process the array in pairs and track the minimum costs
    
    # dp[i][c1][c2] would be too expensive, so we'll use a different approach
    # Instead, we'll process each symmetric pair independently and combine results
    
    # For each symmetric pair (i, j) where j = n-1-i, we need to choose colors (a, b) where a != b
    # We want to minimize the total cost across all pairs
    
    total_cost = 0
    for i in range(n // 2):
        j = n - 1 - i
        
        # Find the minimum cost combination where colors are different
        min_cost = float('inf')
        
        for color_i in range(3):
            for color_j in range(3):
                if color_i != color_j:
                    current_cost = cost[i][color_i] + cost[j][color_j]
                    if current_cost < min_cost:
                        min_cost = current_cost
        
        total_cost += min_cost
    
    return total_cost
