# signature: def postcondition(nums: List[int], x: int, k: int, min_operations_for_k_equal_subarrays: int) -> bool
from typing import List

def postcondition(nums: List[int], x: int, k: int, min_operations_for_k_equal_subarrays: int) -> bool:
    # Validate inputs
    if not (2 <= len(nums) <= 10**5):
        raise ValueError("Invalid input")
    if not all(-10**6 <= num <= 10**6 for num in nums):
        raise ValueError("Invalid input")
    if not (2 <= x <= len(nums)):
        raise ValueError("Invalid input")
    if not (1 <= k <= 15):
        raise ValueError("Invalid input")
    if not (2 <= k * x <= len(nums)):
        raise ValueError("Invalid input")
    if min_operations_for_k_equal_subarrays < 0:
        raise ValueError("Invalid input")
    
    n = len(nums)
    
    # Precompute prefix sums for efficient range sum calculation
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + nums[i]
    
    # Precompute squared prefix sums for efficient range squared sum calculation
    prefix_sq = [0] * (n + 1)
    for i in range(n):
        prefix_sq[i + 1] = prefix_sq[i] + nums[i] * nums[i]
    
    # DP table: dp[i][j] = minimum operations to get j valid subarrays using first i elements
    # Initialize with infinity
    INF = float('inf')
    dp = [[INF] * (k + 1) for _ in range(n + 1)]
    dp[0][0] = 0
    
    for i in range(n + 1):
        for j in range(k + 1):
            if dp[i][j] == INF:
                continue
            
            # Skip current element
            if i + 1 <= n:
                dp[i + 1][j] = min(dp[i + 1][j], dp[i][j])
            
            # Try to form a new subarray starting at position i
            if j < k and i + x <= n:
                start = i
                end = i + x - 1
                
                # Calculate the optimal target value for this subarray
                # The optimal target is the median of the subarray for L1 cost
                # or the mean for L2 cost. For this problem, we need L1 cost.
                subarray = nums[start:end + 1]
                subarray_sorted = sorted(subarray)
                
                # For odd length, median is middle element
                # For even length, any value between the two middle elements gives same cost
                median = subarray_sorted[x // 2]
                
                # Calculate operations needed to make all elements equal to median
                operations = 0
                for num in subarray:
                    operations += abs(num - median)
                
                # Update DP
                dp[i + x][j + 1] = min(dp[i + x][j + 1], dp[i][j] + operations)
    
    # Find the minimum operations needed to get at least k subarrays
    min_required = INF
    for i in range(n + 1):
        if dp[i][k] < min_required:
            min_required = dp[i][k]
    
    # Return True if the given value is at least the minimum required
    return min_operations_for_k_equal_subarrays >= min_required and min_required != INF
