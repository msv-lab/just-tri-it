# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    n = len(nums)
    if n < 2 or len(edges) != n - 1:
        raise ValueError("Invalid input")
    
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length < 1 or length > 1000:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    parent = [-1] * n
    depth = [0] * n
    dist_from_root = [0] * n
    
    stack = [0]
    parent[0] = 0
    while stack:
        node = stack.pop()
        for neighbor, length in graph[node]:
            if parent[neighbor] == -1:
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                dist_from_root[neighbor] = dist_from_root[node] + length
                stack.append(neighbor)
    
    max_length = 0
    min_nodes = float('inf')
    
    for i in range(n):
        for j in range(i, n):
            if i == j:
                path_length = 0
                num_nodes = 1
                if path_length > max_length:
                    max_length = path_length
                    min_nodes = num_nodes
                elif path_length == max_length:
                    min_nodes = min(min_nodes, num_nodes)
                continue
            
            path_nodes = set()
            unique_values = set()
            is_unique = True
            
            node_i, node_j = i, j
            while depth[node_i] > depth[node_j]:
                if nums[node_i] in unique_values:
                    is_unique = False
                    break
                unique_values.add(nums[node_i])
                path_nodes.add(node_i)
                node_i = parent[node_i]
            
            while depth[node_j] > depth[node_i]:
                if nums[node_j] in unique_values:
                    is_unique = False
                    break
                unique_values.add(nums[node_j])
                path_nodes.add(node_j)
                node_j = parent[node_j]
            
            while node_i != node_j:
                if nums[node_i] in unique_values or nums[node_j] in unique_values:
                    is_unique = False
                    break
                unique_values.add(nums[node_i])
                unique_values.add(nums[node_j])
                path_nodes.add(node_i)
                path_nodes.add(node_j)
                node_i = parent[node_i]
                node_j = parent[node_j]
            
            if not is_unique:
                continue
            
            if nums[node_i] in unique_values:
                continue
            
            unique_values.add(nums[node_i])
            path_nodes.add(node_i)
            
            path_length = dist_from_root[i] + dist_from_root[j] - 2 * dist_from_root[node_i]
            num_nodes = len(path_nodes)
            
            if path_length > max_length:
                max_length = path_length
                min_nodes = num_nodes
            elif path_length == max_length:
                min_nodes = min(min_nodes, num_nodes)
    
    if max_length == 0:
        min_nodes = 1
    
    return [max_length, min_nodes]
