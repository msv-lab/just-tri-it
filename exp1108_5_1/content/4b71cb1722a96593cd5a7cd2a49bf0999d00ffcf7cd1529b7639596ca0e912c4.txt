# signature: def minOperations(nums: List[int], x: int, k: int) -> int
from typing import List

def minOperations(nums: List[int], x: int, k: int) -> int:
    n = len(nums)
    
    # Validate constraints
    if n < 2 or n > 10**5:
        raise ValueError("Invalid input")
    if x < 2 or x > n:
        raise ValueError("Invalid input")
    if k < 1 or k > 15:
        raise ValueError("Invalid input")
    if k * x > n:
        raise ValueError("Invalid input")
    if any(num < -10**6 or num > 10**6 for num in nums):
        raise ValueError("Invalid input")
    
    # Precompute prefix sums for efficient range sum calculation
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + nums[i]
    
    # Precompute squared prefix sums for efficient range squared sum calculation
    prefix_sq = [0] * (n + 1)
    for i in range(n):
        prefix_sq[i + 1] = prefix_sq[i] + nums[i] * nums[i]
    
    # DP table: dp[i][j] = min operations to get j subarrays ending at or before position i
    INF = float('inf')
    dp = [[INF] * (k + 1) for _ in range(n + 1)]
    
    # Base case: 0 subarrays require 0 operations
    for i in range(n + 1):
        dp[i][0] = 0
    
    # Fill DP table
    for j in range(1, k + 1):
        for i in range(j * x, n + 1):
            # Option 1: Don't use current position as the end of j-th subarray
            dp[i][j] = min(dp[i][j], dp[i - 1][j])
            
            # Option 2: Use positions [i-x, i-1] as the j-th subarray
            if i >= x:
                # Calculate the optimal target value for this subarray
                # The optimal target is the median (for odd x) or average of two medians (for even x)
                # But for minimum operations, the optimal is the median
                subarray = nums[i - x:i]
                sorted_sub = sorted(subarray)
                
                # For odd length, median is the middle element
                # For even length, any value between the two middle elements gives same cost
                if x % 2 == 1:
                    target = sorted_sub[x // 2]
                else:
                    # For even length, we can choose either middle element
                    target1 = sorted_sub[x // 2 - 1]
                    target2 = sorted_sub[x // 2]
                    # Calculate cost for both and take minimum
                    cost1 = sum(abs(num - target1) for num in subarray)
                    cost2 = sum(abs(num - target2) for num in subarray)
                    target = target1 if cost1 <= cost2 else target2
                
                # Calculate operations needed for this subarray
                operations = 0
                for num in subarray:
                    operations += abs(num - target)
                
                # Update DP
                dp[i][j] = min(dp[i][j], dp[i - x][j - 1] + operations)
    
    # Find minimum operations to get at least k subarrays
    result = INF
    for i in range(k * x, n + 1):
        result = min(result, dp[i][k])
    
    if result == INF:
        raise ValueError("Invalid input")
    
    # Add 1 to the final answer as requested
    return result + 1
