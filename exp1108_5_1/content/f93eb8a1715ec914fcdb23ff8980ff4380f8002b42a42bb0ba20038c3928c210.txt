# signature: def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool
from typing import List
from collections import defaultdict, deque

def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool:
    n = len(nums)
    
    # Validate input constraints
    if n < 2 or n > 50000:
        raise ValueError("Invalid input")
    if len(edges) != n - 1:
        raise ValueError("Invalid input")
    if len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length < 1 or length > 1000:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Validate nums
    if len(nums) != n:
        raise ValueError("Invalid input")
    for num in nums:
        if num < 0 or num > 50000:
            raise ValueError("Invalid input")
    
    # Build tree structure with parent pointers
    parent = [-1] * n
    depth = [0] * n
    dist_from_root = [0] * n
    
    # BFS to build tree and calculate distances
    queue = deque([0])
    parent[0] = -1
    depth[0] = 0
    dist_from_root[0] = 0
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if neighbor != parent[node]:
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                dist_from_root[neighbor] = dist_from_root[node] + length
                queue.append(neighbor)
    
    # Find actual longest special path
    actual_max_length = 0
    actual_min_nodes = float('inf')
    
    # For each node as potential starting point, DFS to find special paths
    for start in range(n):
        stack = [(start, {nums[start]}, 0, 1)]  # (node, values_seen, path_length, node_count)
        
        while stack:
            node, values, path_len, node_count = stack.pop()
            
            # Update global maximum
            if path_len > actual_max_length:
                actual_max_length = path_len
                actual_min_nodes = node_count
            elif path_len == actual_max_length and path_len > 0:
                actual_min_nodes = min(actual_min_nodes, node_count)
            
            # Explore children
            for child, edge_len in graph[node]:
                if child != parent[node]:  # Ensure we're going downward
                    if nums[child] not in values:
                        new_values = values.copy()
                        new_values.add(nums[child])
                        stack.append((child, new_values, path_len + edge_len, node_count + 1))
    
    # Handle case where no path was found (shouldn't happen as single node paths exist)
    if actual_max_length == 0:
        actual_max_length = 0
        actual_min_nodes = 1
    
    # Check if the provided info matches our calculated values
    provided_length = longest_special_path_info[0]
    provided_min_nodes = longest_special_path_info[1]
    
    return provided_length == actual_max_length and provided_min_nodes == actual_min_nodes
