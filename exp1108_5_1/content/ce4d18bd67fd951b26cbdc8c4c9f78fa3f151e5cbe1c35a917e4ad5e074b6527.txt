# signature: def minOperations(nums: List[int], x: int, k: int) -> int
# test name: test_minimum_array_length_maximum_k
def test_minimum_array_length_maximum_k():
    nums = [1, 2, 3, 4]
    x = 2
    k = 2  # Maximum k for length 4: 4/2 = 2
    result = minOperations(nums, x, k)
    # Need to make two pairs equal: best pairs are (1,2) and (3,4)
    # Cost: |1-1.5| + |2-1.5| + |3-3.5| + |4-3.5| = 0.5+0.5+0.5+0.5 = 2
    # Since operations are integer, cost = 2
    assert result == 2
