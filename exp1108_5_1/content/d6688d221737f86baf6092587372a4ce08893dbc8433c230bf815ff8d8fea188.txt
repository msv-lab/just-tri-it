# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    if not edges or len(nums) != len(edges) + 1:
        raise ValueError("Invalid input")
    
    n = len(nums)
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    parent = [-1] * n
    depth = [0] * n
    dist_from_root = [0] * n
    
    stack = [0]
    parent[0] = 0
    while stack:
        node = stack.pop()
        for neighbor, length in graph[node]:
            if parent[neighbor] == -1:
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                dist_from_root[neighbor] = dist_from_root[node] + length
                stack.append(neighbor)
    
    max_length = 0
    min_nodes = float('inf')
    
    for start in range(n):
        stack = [(start, 0, {nums[start]})]
        while stack:
            node, current_length, visited = stack.pop()
            if current_length > max_length:
                max_length = current_length
                min_nodes = 1
            elif current_length == max_length:
                min_nodes = min(min_nodes, 1)
            
            for neighbor, length in graph[node]:
                if neighbor == parent[node]:
                    continue
                if nums[neighbor] in visited:
                    continue
                new_visited = visited.copy()
                new_visited.add(nums[neighbor])
                new_length = current_length + length
                stack.append((neighbor, new_length, new_visited))
    
    if max_length == 0:
        min_nodes = 1
    
    return [max_length, min_nodes]
