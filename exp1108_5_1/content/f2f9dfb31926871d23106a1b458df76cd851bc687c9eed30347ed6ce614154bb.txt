from typing import List
from collections import defaultdict, deque

def longestSpecialPath_inv_nums(longest_special_path_info: List[int], edges: List[List[int]]) -> List[int]:
    if len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    target_length, target_min_nodes = longest_special_path_info
    
    if target_length < 1 or target_min_nodes < 1:
        raise ValueError("Invalid input")
    
    n = len(edges) + 1
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Find parent-child relationships by doing BFS from root 0
    parent = [-1] * n
    children = defaultdict(list)
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append(neighbor)
                queue.append(neighbor)
    
    # Strategy: Create a path of length target_length with target_min_nodes
    # We'll create a simple increasing sequence along a path
    
    nums = [0] * n
    
    if target_length == 1:
        # Single node path
        for i in range(n):
            nums[i] = i
        return nums
    
    # Find a path with enough nodes
    path_nodes = []
    
    def find_path(current, path):
        nonlocal path_nodes
        if path_nodes:
            return
        
        if len(path) == target_min_nodes:
            path_nodes = path.copy()
            return
        
        for child in children[current]:
            path.append(child)
            find_path(child, path)
            path.pop()
    
    find_path(0, [0])
    
    if not path_nodes:
        # If we can't find a path with target_min_nodes, create one by extending
        # We need at least target_min_nodes in the path
        path_nodes = [0]
        current = 0
        while len(path_nodes) < target_min_nodes and children[current]:
            current = children[current][0]
            path_nodes.append(current)
    
    # Assign values to create the special path
    # Use consecutive integers to ensure uniqueness
    for i, node in enumerate(path_nodes):
        nums[node] = i
    
    # Fill remaining nodes with values that won't create longer paths
    # Use values starting from target_min_nodes to avoid conflicts
    next_val = target_min_nodes
    for i in range(n):
        if nums[i] == 0 and i != 0:  # Node not assigned yet and not root
            nums[i] = next_val
            next_val += 1
    
    return nums
