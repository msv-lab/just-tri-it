# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict, deque

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    n = len(nums)
    
    # Validate inputs
    if n < 2:
        raise ValueError("Invalid input")
    if len(edges) != n - 1:
        raise ValueError("Invalid input")
    for edge in edges:
        if len(edge) != 3:
            raise ValueError("Invalid input")
        u, v, length = edge
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Find root (node 0) and build parent-child relationships
    parent = [-1] * n
    depth = [0] * n
    dist_from_root = [0] * n
    
    # BFS to build tree structure from root
    queue = deque([0])
    visited = [False] * n
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                dist_from_root[neighbor] = dist_from_root[node] + length
                queue.append(neighbor)
    
    # DFS to find longest special paths
    max_length = 0
    min_nodes_for_max_length = float('inf')
    
    def dfs(node, path_values, path_nodes, current_dist):
        nonlocal max_length, min_nodes_for_max_length
        
        # Current path from root to node
        path_length = current_dist - dist_from_root[path_nodes[0]] if path_nodes else 0
        
        # Update global maximum
        if path_length > max_length:
            max_length = path_length
            min_nodes_for_max_length = len(path_nodes)
        elif path_length == max_length:
            min_nodes_for_max_length = min(min_nodes_for_max_length, len(path_nodes))
        
        # Explore children
        for child, length in graph[node]:
            if child == parent[node]:
                continue
            
            child_value = nums[child]
            
            if child_value in path_values:
                # Can't include this child in current path due to duplicate value
                # Start a new path from this child
                dfs(child, {child_value}, [child], dist_from_root[child])
            else:
                # Can extend current path
                new_path_values = path_values.copy()
                new_path_values.add(child_value)
                new_path_nodes = path_nodes + [child]
                dfs(child, new_path_values, new_path_nodes, current_dist + length)
    
    # Start DFS from root
    dfs(0, {nums[0]}, [0], 0)
    
    # Handle case where no path with positive length exists
    if max_length == 0:
        # All single-node paths have length 0
        min_nodes_for_max_length = 1
    
    return [max_length, min_nodes_for_max_length]
