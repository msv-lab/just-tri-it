# signature: def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool
from typing import List
from collections import defaultdict, deque

def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool:
    n = len(nums)
    
    # Validate input constraints
    if n < 2 or n > 50000:
        raise ValueError("Invalid input")
    if len(edges) != n - 1:
        raise ValueError("Invalid input")
    if len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    # Build adjacency list for the tree
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length < 1 or length > 1000:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Validate nums array
    if len(nums) != n:
        raise ValueError("Invalid input")
    for num in nums:
        if num < 0 or num > 50000:
            raise ValueError("Invalid input")
    
    # Build the tree structure (rooted at 0)
    parent = [-1] * n
    depth = [0] * n
    dist_from_root = [0] * n
    
    # BFS to build tree structure and calculate distances
    queue = deque([0])
    parent[0] = -1
    depth[0] = 0
    dist_from_root[0] = 0
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if neighbor != parent[node]:
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                dist_from_root[neighbor] = dist_from_root[node] + length
                queue.append(neighbor)
    
    # Function to check if a path is valid (all unique values)
    def is_valid_path(nodes):
        values = set()
        for node in nodes:
            if nums[node] in values:
                return False
            values.add(nums[node])
        return True
    
    # Find all special paths and their lengths
    max_length = 0
    min_nodes_for_max_length = float('inf')
    
    # For each possible path (ancestor to descendant)
    for i in range(n):
        # Start DFS from node i to find all descendants
        stack = [(i, [i], {nums[i]})]
        
        while stack:
            current, path, values_set = stack.pop()
            
            # Check if current path is special
            path_length = dist_from_root[current] - dist_from_root[path[0]]
            
            if len(path) > 1:  # Path must have at least 2 nodes
                if path_length > max_length:
                    max_length = path_length
                    min_nodes_for_max_length = len(path)
                elif path_length == max_length:
                    min_nodes_for_max_length = min(min_nodes_for_max_length, len(path))
            
            # Continue to children
            for neighbor, length in graph[current]:
                if neighbor != parent[current]:
                    if nums[neighbor] not in values_set:
                        new_path = path + [neighbor]
                        new_values = values_set | {nums[neighbor]}
                        stack.append((neighbor, new_path, new_values))
    
    # Handle case where no path with >1 node was found
    if max_length == 0:
        # Check single-node paths (path from node to itself)
        for i in range(n):
            if 0 > max_length:
                max_length = 0
                min_nodes_for_max_length = 1
            elif 0 == max_length:
                min_nodes_for_max_length = min(min_nodes_for_max_length, 1)
    
    # Verify the claimed longest_special_path_info
    claimed_length = longest_special_path_info[0]
    claimed_min_nodes = longest_special_path_info[1]
    
    return claimed_length == max_length and claimed_min_nodes == min_nodes_for_max_length
