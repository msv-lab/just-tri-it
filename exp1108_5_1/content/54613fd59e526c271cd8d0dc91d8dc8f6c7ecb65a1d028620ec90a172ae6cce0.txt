# signature: def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool
from typing import List
from collections import defaultdict, deque

def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool:
    n = len(nums)
    
    # Validate input constraints
    if n < 2 or n > 50000:
        raise ValueError("Invalid input")
    if len(edges) != n - 1:
        raise ValueError("Invalid input")
    if len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length < 1 or length > 1000:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Build tree structure (rooted at 0)
    parent = [-1] * n
    depth = [0] * n
    children = defaultdict(list)
    edge_lengths = {}
    
    # BFS to build tree structure
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                children[node].append(neighbor)
                edge_lengths[(node, neighbor)] = length
                edge_lengths[(neighbor, node)] = length
                queue.append(neighbor)
    
    # Find actual longest special path using DFS
    max_path_length = 0
    min_nodes_for_max_length = float('inf')
    
    def dfs(node, current_path_values, current_path_nodes, current_length):
        nonlocal max_path_length, min_nodes_for_max_length
        
        # Check if current path is special (all unique values)
        if len(current_path_values) == len(set(current_path_values)):
            # This is a valid special path
            if current_length > max_path_length:
                max_path_length = current_length
                min_nodes_for_max_length = len(current_path_nodes)
            elif current_length == max_path_length:
                min_nodes_for_max_length = min(min_nodes_for_max_length, len(current_path_nodes))
        
        # Explore children
        for child in children[node]:
            child_value = nums[child]
            edge_len = edge_lengths[(node, child)]
            
            # Try extending the path with this child
            new_path_values = current_path_values + [child_value]
            new_path_nodes = current_path_nodes + [child]
            new_length = current_length + edge_len
            
            dfs(child, new_path_values, new_path_nodes, new_length)
    
    # Start DFS from root and all possible starting points
    for start_node in range(n):
        dfs(start_node, [nums[start_node]], [start_node], 0)
    
    # Check if the provided info matches the actual values
    claimed_length = longest_special_path_info[0]
    claimed_min_nodes = longest_special_path_info[1]
    
    return claimed_length == max_path_length and claimed_min_nodes == min_nodes_for_max_length
