from typing import List
from collections import defaultdict, deque

def longestSpecialPath_inv_nums(longest_special_path_info: List[int], edges: List[List[int]]) -> List[int]:
    if not longest_special_path_info or len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    target_length, target_min_nodes = longest_special_path_info
    
    if target_length < 1 or target_min_nodes < 1 or target_min_nodes > target_length:
        raise ValueError("Invalid input")
    
    n = len(edges) + 1
    
    # Build the tree
    graph = defaultdict(list)
    for u, v, length in edges:
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Find parent-child relationships by doing BFS from root 0
    parent = [-1] * n
    children = defaultdict(list)
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                children[node].append(neighbor)
                queue.append(neighbor)
    
    # Case 1: If target_length == 1, we need a path with exactly 1 node
    if target_length == 1:
        nums = [0] * n
        return nums
    
    # Case 2: If target_length == target_min_nodes, we need a simple increasing sequence
    elif target_length == target_min_nodes:
        nums = [0] * n
        # Create a path of length target_length with unique values
        path_nodes = []
        stack = [0]
        while stack and len(path_nodes) < target_length:
            node = stack.pop()
            path_nodes.append(node)
            # Add children in reverse order to maintain DFS order
            for child in reversed(children[node]):
                if len(path_nodes) < target_length:
                    stack.append(child)
        
        # Assign unique values to path nodes
        for i, node in enumerate(path_nodes):
            nums[node] = i
        
        # Assign values to remaining nodes (can be duplicates)
        for i in range(n):
            if nums[i] == 0 and i not in path_nodes:
                nums[i] = 0
        
        return nums
    
    # Case 3: General case - we need a path where some values repeat to make min_nodes < length
    else:
        nums = [0] * n
        
        # Create the main path with target_length nodes
        path_nodes = []
        stack = [0]
        while stack and len(path_nodes) < target_length:
            node = stack.pop()
            path_nodes.append(node)
            for child in reversed(children[node]):
                if len(path_nodes) < target_length:
                    stack.append(child)
        
        # We need target_min_nodes unique values in a path of length target_length
        # This means we need (target_length - target_min_nodes) duplicates
        
        # Assign values to create the required number of duplicates
        current_val = 0
        used_values = set()
        duplicate_positions = target_length - target_min_nodes
        
        for i, node in enumerate(path_nodes):
            if i < target_min_nodes:
                nums[node] = current_val
                used_values.add(current_val)
                current_val += 1
            else:
                # Reuse a value from earlier in the path to create a duplicate
                nums[node] = list(used_values)[i % target_min_nodes]
        
        # Fill remaining nodes with any values
        for i in range(n):
            if nums[i] == 0 and i not in path_nodes:
                nums[i] = 0
        
        return nums
