# signature: def minOperations(nums: List[int], x: int, k: int) -> int
from typing import List

def minOperations(nums: List[int], x: int, k: int) -> int:
    n = len(nums)
    
    # Validate inputs
    if n < 2 or n > 10**5:
        raise ValueError("Invalid input")
    if any(num < -10**6 or num > 10**6 for num in nums):
        raise ValueError("Invalid input")
    if x < 2 or x > n:
        raise ValueError("Invalid input")
    if k < 1 or k > 15:
        raise ValueError("Invalid input")
    if k * x > n:
        raise ValueError("Invalid input")
    
    # Precompute prefix sums for efficient range sum calculation
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + nums[i]
    
    # DP table: dp[i][j] = min operations to get j subarrays using first i+x-1 elements
    # We need k subarrays, so dp dimension is k+1
    INF = float('inf')
    dp = [[INF] * (k + 1) for _ in range(n + 1)]
    dp[0][0] = 0
    
    for i in range(n - x + 1):
        for j in range(k + 1):
            if dp[i][j] == INF:
                continue
            
            # Option 1: Skip current position
            dp[i + 1][j] = min(dp[i + 1][j], dp[i][j])
            
            # Option 2: Start a subarray at position i (if we have space for more subarrays)
            if j < k and i + x <= n:
                # Calculate the target value that minimizes operations for this subarray
                # The optimal target is the median of the subarray for L1 distance minimization
                subarray = nums[i:i + x]
                subarray.sort()
                median = subarray[x // 2]
                
                # Calculate operations needed to make all elements equal to median
                operations = 0
                for num in subarray:
                    operations += abs(num - median)
                
                # Update DP state
                dp[i + x][j + 1] = min(dp[i + x][j + 1], dp[i][j] + operations)
    
    # Find the minimum operations to get at least k subarrays
    result = INF
    for i in range(n + 1):
        if dp[i][k] < result:
            result = dp[i][k]
    
    if result == INF:
        raise ValueError("Invalid input")
    
    return result + 1
