# signature: def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]
from typing import List
from collections import defaultdict, deque

def longestSpecialPath(edges: List[List[int]], nums: List[int]) -> List[int]:
    n = len(nums)
    if n < 2 or len(edges) != n - 1:
        raise ValueError("Invalid input")
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length <= 0:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Build tree structure with parent-child relationships (root at 0)
    tree = defaultdict(list)
    parent = [-1] * n
    dist_from_root = [0] * n
    
    # BFS to establish parent-child relationships and distances from root
    queue = deque([0])
    visited = [False] * n
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                dist_from_root[neighbor] = dist_from_root[node] + length
                tree[node].append((neighbor, length))
                queue.append(neighbor)
    
    # Global variables to track the best results
    max_length = 0
    min_nodes_for_max = float('inf')
    
    def dfs(node, path_values, path_nodes, current_length):
        nonlocal max_length, min_nodes_for_max
        
        # Check if current path is special (all unique values)
        if len(path_values) == len(set(path_values)):
            # Update global maximum
            if current_length > max_length:
                max_length = current_length
                min_nodes_for_max = len(path_nodes)
            elif current_length == max_length:
                min_nodes_for_max = min(min_nodes_for_max, len(path_nodes))
        
        # Explore children
        for child, length in tree[node]:
            # Add child to path
            path_values.append(nums[child])
            path_nodes.append(child)
            
            dfs(child, path_values, path_nodes, current_length + length)
            
            # Backtrack
            path_values.pop()
            path_nodes.pop()
    
    # Start DFS from root (node 0)
    dfs(0, [nums[0]], [0], 0)
    
    # Handle case where no special path found (shouldn't happen as single nodes are valid)
    if max_length == 0:
        min_nodes_for_max = 1
    
    result = [max_length, min_nodes_for_max]
    
    # Append 1 to the result as requested
    result.append(1)
    return result
