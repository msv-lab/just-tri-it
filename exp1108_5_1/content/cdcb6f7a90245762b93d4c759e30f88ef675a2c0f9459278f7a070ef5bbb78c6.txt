# signature: def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool
from typing import List
from collections import defaultdict, deque

def postcondition(edges: List[List[int]], nums: List[int], longest_special_path_info: List[int]) -> bool:
    n = len(nums)
    
    # Validate input constraints
    if n < 2 or n > 50000:
        raise ValueError("Invalid input")
    if len(edges) != n - 1:
        raise ValueError("Invalid input")
    if len(longest_special_path_info) != 2:
        raise ValueError("Invalid input")
    
    # Build adjacency list
    graph = defaultdict(list)
    for u, v, length in edges:
        if u < 0 or u >= n or v < 0 or v >= n or length < 1 or length > 1000:
            raise ValueError("Invalid input")
        graph[u].append((v, length))
        graph[v].append((u, length))
    
    # Validate nums
    if len(nums) != n:
        raise ValueError("Invalid input")
    for num in nums:
        if num < 0 or num > 50000:
            raise ValueError("Invalid input")
    
    # Build tree structure with parent-child relationships
    parent = [-1] * n
    depth = [0] * n
    children = defaultdict(list)
    
    # BFS to build tree structure from root 0
    visited = [False] * n
    queue = deque([0])
    visited[0] = True
    
    while queue:
        node = queue.popleft()
        for neighbor, length in graph[node]:
            if not visited[neighbor]:
                visited[neighbor] = True
                parent[neighbor] = node
                depth[neighbor] = depth[node] + 1
                children[node].append(neighbor)
                queue.append(neighbor)
    
    # DFS to find actual longest special path
    max_length = 0
    min_nodes_for_max_length = float('inf')
    
    def dfs(node, path_values, path_length, path_nodes):
        nonlocal max_length, min_nodes_for_max_length
        
        current_value = nums[node]
        if current_value in path_values:
            return
        
        # Update path info
        new_path_values = path_values | {current_value}
        new_path_nodes = path_nodes + 1
        
        # Check if this is a special path
        if new_path_nodes > 1:  # At least 2 nodes for a path
            if new_path_length > max_length:
                max_length = new_path_length
                min_nodes_for_max_length = new_path_nodes
            elif new_path_length == max_length:
                min_nodes_for_max_length = min(min_nodes_for_max_length, new_path_nodes)
        
        # Continue DFS to children
        for child in children[node]:
            dfs(child, new_path_values, new_path_length, new_path_nodes)
    
    # Start DFS from each node as potential starting point
    for start_node in range(n):
        dfs(start_node, set(), 0, 0)
    
    # Handle case where no special path with at least 2 nodes exists
    if max_length == 0:
        # Check if single node paths are considered (problem says path may start and end at same node)
        # For single node paths, length would be 0, nodes would be 1
        if longest_special_path_info[0] == 0 and longest_special_path_info[1] == 1:
            return True
        return False
    
    # Check if the claimed info matches our calculated values
    claimed_length, claimed_min_nodes = longest_special_path_info
    
    return claimed_length == max_length and claimed_min_nodes == min_nodes_for_max_length
